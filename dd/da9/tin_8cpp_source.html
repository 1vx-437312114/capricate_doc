235 void Basic_TMesh::init(const Triangle *t0, const bool keep_reference)
236 {
237  info=NULL;
238  List todo(t0), st, sv, se;
239  Node *n;
240  Triangle *t, *nt;
241  Vertex *v, *nv;
242  Edge *e, *ne;
243 
244  t=(Triangle *)t0;
245  MARK_VISIT2(t);
246 
247  while (todo.numels())
248  {
249  t = (Triangle *)todo.popHead();
250  st.appendHead(t);
251  nt=t->t1(); if (nt != NULL && !IS_VISITED2(nt)) {MARK_VISIT2(nt); todo.appendHead(nt);}
252  nt=t->t2(); if (nt != NULL && !IS_VISITED2(nt)) {MARK_VISIT2(nt); todo.appendHead(nt);}
253  nt=t->t3(); if (nt != NULL && !IS_VISITED2(nt)) {MARK_VISIT2(nt); todo.appendHead(nt);}
254  }
255 
256  FOREACHVTTRIANGLE((&st), t, n)
257  {
258  UNMARK_VISIT2(t);
259  e = t->e1; if (!IS_VISITED2(e)) {MARK_VISIT2(e); se.appendHead(e);}
260  e = t->e2; if (!IS_VISITED2(e)) {MARK_VISIT2(e); se.appendHead(e);}
261  e = t->e3; if (!IS_VISITED2(e)) {MARK_VISIT2(e); se.appendHead(e);}
262  v = t->v1(); if (!IS_VISITED2(v)) {MARK_VISIT2(v); sv.appendHead(v);}
263  v = t->v2(); if (!IS_VISITED2(v)) {MARK_VISIT2(v); sv.appendHead(v);}
264  v = t->v3(); if (!IS_VISITED2(v)) {MARK_VISIT2(v); sv.appendHead(v);}
265  }
266 
267  FOREACHVVVERTEX((&sv), v, n)
268  {UNMARK_VISIT2(v); nv=newVertex(v); V.appendTail(nv); v->info = nv;}
269 
270  FOREACHVEEDGE((&se), e, n)
271  {UNMARK_VISIT2(e); ne=newEdge((Vertex *)e->v1->info, (Vertex *)e->v2->info); E.appendTail(ne); e->info = ne;}
272 
273  FOREACHVTTRIANGLE((&st), t, n)
274  {nt=newTriangle((Edge *)t->e1->info,(Edge *)t->e2->info,(Edge *)t->e3->info); T.appendTail(nt); t->info = nt;}
275 
276  FOREACHVVVERTEX((&sv), v, n) ((Vertex *)v->info)->e0 = (Edge *)v->e0->info;
277 
278  FOREACHVEEDGE((&se), e, n)
279  {((Edge *)e->info)->t1 = (e->t1)?((Triangle *)e->t1->info):(NULL); ((Edge *)e->info)->t2 = (e->t2)?((Triangle *)e->t2->info):(NULL);}
280 
281  if (!keep_reference)
282  {
283  FOREACHVVVERTEX((&sv), v, n) v->info = NULL;
284  FOREACHVEEDGE((&se), e, n) e->info = NULL;
285  FOREACHVTTRIANGLE((&st), t, n) t->info = NULL;
286  }
287 
288  eulerUpdate();
289 }

481  if (e2->isIsolated() && e3->isIsolated()) v2->e0 = NULL;
482  if (e3->isIsolated() && e1->isIsolated()) v3->e0 = NULL;
483  if (e1->isIsolated()) e1->v1 = e1->v2 = NULL;
484  if (e2->isIsolated()) e2->v1 = e2->v2 = NULL;
485  if (e3->isIsolated()) e3->v1 = e3->v2 = NULL;
