285  j=0; FOREACHVVVERTEX((&(ovbs)), v, n)

288  }
289  delete [] ovps; //free(ovps); // AMF_CHANGE - since IMAT-STL 2.4-2
290 
291  return nt;
292 }

711 // Joins the two boundary vertices gv and gw through an edge. A pair of triangles is
712 // added to properly change the topology of the mesh.
713 // On success, the return value is the new edge connecting the two vertices.
714 // NULL is returned on failure.
715 // Failure occurs if gv and gw are not both on boundary.
716 // If 'justconnect' is false, the remaining hole is filled with new triangles, unless
717 // gv and gw are contiguous on the boundary loop (failure).
718 // If 'justconnect' is true, gv and gw must not belong to the same boundary loop (failure).
719 // If 'refine', the patching triangles are refined to reproduce neighboring density.
720 
721 Edge *Basic_TMesh::joinBoundaryLoops(Vertex *gv, Vertex *gw, bool justconnect, bool refine)
722 {
723  Vertex *v, *gvn, *gwn;
724  Edge *e, *gve, *gwe;
725  Triangle *t;
726  Node *n;
727  double tl1 = 0.0, tl2 = 0.0, pl1, pl2;
728 
729  if (gv == NULL || gw == NULL || !gv->isOnBoundary() || !gw->isOnBoundary()) return NULL;
730 
731  FOREACHVERTEX(v, n) UNMARK_VISIT(v);
732  deselectTriangles();
733 
734  v = gv;
735  if (!justconnect)
736  {
737  do { v = v->nextOnBoundary(); if (v == gw) return NULL; } while (v != gv);
738  } else
739  {
740  gvn = gv->nextOnBoundary(); gwn = gv->prevOnBoundary();
741  if (gw == gvn || gw == gwn) return NULL;
742  if (gw == gvn->nextOnBoundary())
743  {
744  t = EulerEdgeTriangle(gvn->prevBoundaryEdge(), gvn->nextBoundaryEdge()); MARK_VISIT(t); return t->oppositeEdge(gvn);
745  }
746  if (gw == gwn->prevOnBoundary())
747  {
748  t = EulerEdgeTriangle(gwn->prevBoundaryEdge(), gwn->nextBoundaryEdge()); MARK_VISIT(t); return t->oppositeEdge(gwn);
749  }
750  }
751 
752  gve = gv->prevBoundaryEdge();
753  gvn = gve->oppositeVertex(gv);
754  gwe = gw->nextBoundaryEdge();
755  gwn = gwe->oppositeVertex(gw);
756 
757  Edge *je = CreateEdge(gv, gw);
758  Edge *je1 = CreateEdge(gv, gwn);
759  Edge *je2 = CreateEdge(gwn, gvn);
760 
761  t = CreateTriangle(je, gwe, je1); MARK_VISIT(t);
762  t = CreateTriangle(je1, je2, gve); MARK_VISIT(t);
763 
764  if (justconnect) return je;
765 
766  v = gv; do { e = v->nextBoundaryEdge(); v = e->oppositeVertex(v); tl1 += e->length(); } while (v != gv);
767  v = gw; do { e = v->nextBoundaryEdge(); v = e->oppositeVertex(v); tl2 += e->length(); } while (v != gw);
768  pl1 = tl1; pl2 = tl2;
769 
770  double c1, c2;
771 
772  e = je;
773  while (e->isOnBoundary())
774  {
775  gv = (e->t2 != NULL) ? (e->v2) : (e->v1); gve = gv->nextBoundaryEdge();
776  gw = (e->t1 != NULL) ? (e->v2) : (e->v1); gwe = gw->prevBoundaryEdge();
777  c1 = fabs((pl1 - gve->length())*tl2 - pl2*tl1);
778  c2 = fabs((pl2 - gwe->length())*tl1 - pl1*tl2);
779  if (c1<c2)
780  {
781  t = EulerEdgeTriangle(e, gve); MARK_VISIT(t);
782  pl1 -= gve->length();
783  e = t->nextEdge(gve);
784  } else
785  {
786  t = EulerEdgeTriangle(gwe, e); MARK_VISIT(t);
787  pl2 -= gwe->length();
788  e = t->prevEdge(gwe);
789  }
790  }
791 
792  if (refine) refineSelectedHolePatches();
793 
794  return je;
795 }
