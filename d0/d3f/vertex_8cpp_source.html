320 bool Vertex::isDoubleFlat(Edge **e1, Edge **e2) const
321 {

341 //// Unlinks the vertex if it is either Flat() or DoubleFlat(). On success, the function returns TRUE,
342 //// the vertex neighborhood is retriangulated, and the geometric realization does not change.
343 bool Vertex::removeIfRedundant(bool check_neighborhood)
344 {
345  Edge *e, *e1 = NULL, *e2 = NULL;
346  if (!isDoubleFlat(&e1, &e2)) return false;
347 
348  Node *n;
349  Vertex *vo1, *vo2;
350  List *ve;
351 
352  if (check_neighborhood)
353  {
354  ve = VT();
355  Triangle *t;
356  FOREACHVTTRIANGLE(ve, t, n) if (t->isExactlyDegenerate()) { delete ve; return false; }
357  delete ve;
358  ve = VE();
359  FOREACHVEEDGE(ve, e, n) if (e->overlaps()) { delete ve; return false; }
360  } else ve = VE();
361 
362  if (e1 != NULL) ve->removeNode(e1); // means isDoubleFlat()
363  if (e2 != NULL) ve->removeNode(e2); // means isDoubleFlat()
364  if (e2 != NULL && *e1->oppositeVertex(this) == *e2->oppositeVertex(this)) { delete ve; return false; }
365 
366  while (1)
367  {
368  FOREACHVEEDGE(ve, e, n)
369  {
370  vo1 = e->t1->oppositeVertex(e);
371  vo2 = e->t2->oppositeVertex(e);
372  if (!e->v1->exactSameSideOnPlane(e->v2, vo1, vo2) && vo1->exactMisalignment(e->v1, vo2) && vo1->exactMisalignment(e->v2, vo2))
373  {
374  if (!e->swap()) { delete ve; return false; }
375  break;
376  }
377  }
378  if (n != NULL) ve->removeCell(n);
379  else break;
380  }
381 
382  if (e1 != NULL) e = e1; // means isDoubleFlat()
383  else if (ve->numels() == 3) e = (Edge *)ve->head()->data;
384  else
385  {
386  FOREACHVEEDGE(ve, e, n)
387  if (!e->t1->oppositeVertex(e)->exactMisalignment(this, e->t2->oppositeVertex(e))) break;
388  if (n == NULL) { delete ve; return false; } // Should not happen...
389  else e = (Edge *)((n == ve->head()) ? (ve->tail()) : (n->prev()))->data;
390  }
391  delete ve;
392 
393  if (e->v1 == this) e->invert();
394  if (e->collapseOnV1() == NULL) return false; // This is a very rare case. Should be treated, but does not hurt too much.
395  else return true;
396 }

442 double Vertex::getAngleForTriangulation() const
443 {
444  Edge *e1 = prevBoundaryEdge();
445  Edge *e2 = nextBoundaryEdge();
446  if (e1 == NULL || e2 == NULL) return DBL_MAX;
447  Triangle *t1 = e1->getBoundaryTriangle();
448  Triangle *t2 = e2->getBoundaryTriangle();
449  Vertex *v1 = e1->oppositeVertex(this);
450  Vertex *v2 = e2->oppositeVertex(this);
451  if ((*v2)==(*v1)) return -2;
452  if (distance(v1)*distance(v2) == 0.0) return -1;
453 
454  double ang = getAngle(v1, v2);
455  if (ang == M_PI) return 3*M_PI;
456  if (ang == 0) return 0;
457 
458  Edge e3(v1,v2);
459  Triangle t(e1,e2,&e3);
460  double da1 = t.getDAngle(t1);
461  double da2 = t.getDAngle(t2);
462 
463  if (da1==M_PI && da2==M_PI) return (DBL_MAX/2.0);
464  if (da1==M_PI || da2==M_PI) return (DBL_MAX/4.0);
465 
466  return da1+da2+ang;
467 }

473 {

475  Edge *e2 = nextBoundaryEdge();

479  Point p, p1, p2;

481  p2.setValue(v2);
482  p.setValue(this);
483  p.project(nor);
484  p1.project(nor);
485  p2.project(nor);
486  if (p.distance(p1)*p.distance(p2) == 0.0)

558 int Vertex::zip(const bool check_geom)
559 {
560  Node *n;
561  Edge *e;
562 
563  List *ve = VE();
564  Edge *be1 = (Edge *)ve->head()->data;
565  Edge *be2 = (Edge *)ve->tail()->data;
566  delete(ve);
567  if (!be1->isOnBoundary() || !be2->isOnBoundary()) return 0;

570 
571  if (check_geom && ((*ov1)!=(*ov2))) return 0;
572

575  ve = ov2->VE();
576  FOREACHVEEDGE(ve, e, n) e->replaceVertex(ov2, ov1);
577  delete(ve);
578  ov2->e0 = NULL;
579  }

585  e0 = ov1->e0 = be1;
586  return 1+ov1->zip(check_geom);

645 Edge *Vertex::inverseCollapse(Vertex *v2, Edge *e, Edge *e1, Edge *e2, Edge *e3, Edge *e4, Triangle *t1, Triangle *t2)
646 {
647  Triangle *ta1, *ta4;
648  Node *n;
649  Edge *f;
650 
651  ta1 = e2->rightTriangle(this);
652  ta4 = e3->leftTriangle(this);
653 
654  List *ve = VE();
655  FOREACHVEEDGE(ve, f, n) if (f == e3) break;
656  FOREACHNODECIRCULAR((*ve), n, n)

658  f = ((Edge *)n->data);
659  if (f == e2) break;
660  else f->replaceVertex(this, v2);
661  }
662  delete(ve);
663

665  e1->v1 = v2; e1->v2 = e2->oppositeVertex(this);
666  e4->v1 = v2; e4->v2 = e3->oppositeVertex(this);
667  t1->e1 = e; t1->e2 = e1; t1->e3 = e2;
668  t2->e1 = e; t2->e2 = e3; t2->e3 = e4;
669 
670  e->t1 = t1; e->t2 = t2;
671  e2->replaceTriangle(ta1, t1);
