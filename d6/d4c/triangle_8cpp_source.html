207 coord Triangle::pointTriangleSquaredDistance(const Point *p, Edge **closest_edge, Vertex **closest_vertex) const
208 {
209  Vertex *va = v1(), *vb = v2(), *vc = v3();
210  Point n(((*va)-(*vb))&((*vb)-(*vc)));
211  if (n.x == 0 && n.y == 0 && n.z == 0) return -1.0;
212 
213  coord d1 = ((((*va)-(*vb))&((*vb)-(*p)))*n);
214  coord d2 = ((((*vb)-(*vc))&((*vc)-(*p)))*n);
215  coord d3 = ((((*vc)-(*va))&((*va)-(*p)))*n);
216 
217  if (d1 > 0 && d2 > 0 && d3 > 0) // Closest point in inner triangle
218  {
219  if (closest_edge != NULL) *closest_edge = NULL;
220  if (closest_vertex != NULL) *closest_vertex = NULL;
221  return squaredDistanceFromPoint(p);
222  }
223 
224  if (d2 < 0) { va = vb; vb = vc; if (closest_edge != NULL) *closest_edge = e3; }
225  else if (d3 < 0) { vb = va; va = vc; if (closest_edge != NULL) *closest_edge = e1; }
226  else if (closest_edge != NULL) *closest_edge = e2;
227 
228  Point i(p->projection(va,vb));
229  Point p1(i-(*va)); Point p2(i-(*vb));
230 
231  if (p1*p2 < 0) // Closest point on interior of one edge
232  {
233  return i.squaredDistance(p);
234  }
235 
236  d1=p1.squaredLength(); d2=p2.squaredLength();
237  if (d1 < d2) { if (closest_vertex != NULL) *closest_vertex = va; return p->squaredDistance(va); }
238  else { if (closest_vertex != NULL) *closest_vertex = vb; return p->squaredDistance(vb); }
239 }

305 bool Triangle::intersects(const Triangle *t2, bool justproper) const
306 {
307  Vertex *v11, *v12, *v13, *v21, *v22, *v23;
308 
309  if (justproper)
310  {
311  // This works for non-degenerate triangles. Not sure it will work for degeneracies too.
312  v11 = v1(); v12 = v2(); v13 = v3();
313  v21 = t2->v1(); v22 = t2->v2(); v23 = t2->v3();
314  Vertex *eq1 = ((*v11) == (*v21)) ? (v21) : (((*v11) == (*v22)) ? (v22) : (((*v11) == (*v23)) ? (v23) : (NULL)));
315  Vertex *eq2 = ((*v12) == (*v21)) ? (v21) : (((*v12) == (*v22)) ? (v22) : (((*v12) == (*v23)) ? (v23) : (NULL)));
316  Vertex *eq3 = ((*v13) == (*v21)) ? (v21) : (((*v13) == (*v22)) ? (v22) : (((*v13) == (*v23)) ? (v23) : (NULL)));
317  if (eq1 && eq2 && eq3) return false; // Triangles coincide
318  Edge *ce1 = NULL, *ce2 = NULL;
319  if (eq1 && eq2) { ce1 = e2; ce2 = (t2->e1->hasVertices(eq1, eq2)) ? (t2->e1) : ((t2->e2->hasVertices(eq1, eq2)) ? (t2->e2) : (t2->e3)); }
320  if (eq2 && eq3) { ce1 = e3; ce2 = (t2->e1->hasVertices(eq3, eq2)) ? (t2->e1) : ((t2->e2->hasVertices(eq3, eq2)) ? (t2->e2) : (t2->e3)); }
321  if (eq3 && eq1) { ce1 = e1; ce2 = (t2->e1->hasVertices(eq3, eq1)) ? (t2->e1) : ((t2->e2->hasVertices(eq3, eq1)) ? (t2->e2) : (t2->e3)); }
322  if (ce1)
323  {
324  Vertex *ov = t2->oppositeVertex(ce2);
325  return (ov->exactOrientation(v11, v12, v13) == 0 && ov->exactSameSideOnPlane(oppositeVertex(ce1), ce1->v1, ce1->v2));
326  }
327  Vertex *cv1 = NULL, *cv2 = NULL;
328  if (eq1) { cv1 = v11; cv2 = eq1; }
329  if (eq2) { cv1 = v12; cv2 = eq2; }
330  if (eq3) { cv1 = v13; cv2 = eq3; }
331  if (cv1) // If they share a vertex, intersection occurs if the opposite edge intersect the triangle
332  {
333  Edge *ee1 = oppositeEdge(cv1), *ee2 = t2->oppositeEdge(cv2);
334  return (Point::segmentIntersectsTriangle(ee1->v1, ee1->v2, v21, v22, v23) ||
335  Point::segmentIntersectsTriangle(ee2->v1, ee2->v2, v11, v12, v13));
336  }
337  }
338  else
339  {
340  Edge *ce = commonEdge(t2);
341  if (ce) // If they share an edge, intersection occurs only if t1 and t2 overlap
342  {
343  Vertex *ov = t2->oppositeVertex(ce);
344  return (ov->exactOrientation(v1(), v2(), v3()) == 0 && ov->exactSameSideOnPlane(oppositeVertex(ce), ce->v1, ce->v2));
345  }
346 
347  Vertex *cv = commonVertex(t2);
348  v11 = v1(); v12 = v2(); v13 = v3();
349  v21 = t2->v1(); v22 = t2->v2(); v23 = t2->v3();
350  if (cv) // If they share a vertex, intersection occurs if the opposite edge intersect the triangle
351  {
352  Edge *ee1 = oppositeEdge(cv), *ee2 = t2->oppositeEdge(cv);
353  return (Point::segmentIntersectsTriangle(ee1->v1, ee1->v2, v21, v22, v23) ||
354  Point::segmentIntersectsTriangle(ee2->v1, ee2->v2, v11, v12, v13));
355  }
356  }
357 
358  // Fast reject by bounding box
359  coord mx = MIN(v11->x, MIN(v13->x, v12->x));
360  if (v21->x < mx && v22->x < mx && v23->x < mx) return false;
361  mx = MAX(v11->x, MAX(v13->x, v12->x));
362  if (v21->x > mx && v22->x > mx && v23->x > mx) return false;
363  mx = MIN(v11->y, MIN(v13->y, v12->y));
364  if (v21->y < mx && v22->y < mx && v23->y < mx) return false;
365  mx = MAX(v11->y, MAX(v13->y, v12->y));
366  if (v21->y > mx && v22->y > mx && v23->y > mx) return false;
367  mx = MIN(v11->z, MIN(v13->z, v12->z));
368  if (v21->z < mx && v22->z < mx && v23->z < mx) return false;
369  mx = MAX(v11->z, MAX(v13->z, v12->z));
370  if (v21->z > mx && v22->z > mx && v23->z > mx) return false;
371 
372  // Calculate relative orientations
373  coord o11 = v11->exactOrientation(v21, v22, v23);
374  coord o12 = v12->exactOrientation(v21, v22, v23);
375  coord o13 = v13->exactOrientation(v21, v22, v23);
376  if ((o11>0 && o12>0 && o13>0) || (o11<0 && o12<0 && o13<0)) return false; // t1 above/below t2
377  coord o21 = v21->exactOrientation(v11, v12, v13);
378  coord o22 = v22->exactOrientation(v11, v12, v13);
379  coord o23 = v23->exactOrientation(v11, v12, v13);
380  if ((o21>0 && o22>0 && o23>0) || (o21<0 && o22<0 && o23<0)) return false; // t2 above/below t1
381 
382  if (o11 == 0 && o12 == 0 && o13 == 0) // t1 and t2 are coplanar
383  {
384  if (Point::innerSegmentsCross(v11, v12, v21, v22)) return true;
385  if (Point::innerSegmentsCross(v11, v12, v22, v23)) return true;
386  if (Point::innerSegmentsCross(v11, v12, v23, v21)) return true;
387  if (Point::innerSegmentsCross(v12, v13, v21, v22)) return true;
388  if (Point::innerSegmentsCross(v12, v13, v22, v23)) return true;
389  if (Point::innerSegmentsCross(v12, v13, v23, v21)) return true;
390  if (Point::innerSegmentsCross(v13, v11, v21, v22)) return true;
391  if (Point::innerSegmentsCross(v13, v11, v22, v23)) return true;
392  if (Point::innerSegmentsCross(v13, v11, v23, v21)) return true;
393  return (
394  Point::pointInTriangle(v11, v21, v22, v23) ||
395  Point::pointInTriangle(v12, v21, v22, v23) ||
396  Point::pointInTriangle(v13, v21, v22, v23) ||
397  Point::pointInTriangle(v21, v11, v12, v13) ||
398  Point::pointInTriangle(v22, v11, v12, v13) ||
399  Point::pointInTriangle(v23, v11, v12, v13));
400  }
401  else return (
402  Point::segmentIntersectsTriangle(v11, v12, v21, v22, v23, o11, o12) ||
403  Point::segmentIntersectsTriangle(v12, v13, v21, v22, v23, o12, o13) ||
404  Point::segmentIntersectsTriangle(v13, v11, v21, v22, v23, o13, o11) ||
405  Point::segmentIntersectsTriangle(v21, v22, v11, v12, v13, o21, o22) ||
406  Point::segmentIntersectsTriangle(v22, v23, v11, v12, v13, o22, o23) ||
407  Point::segmentIntersectsTriangle(v23, v21, v11, v12, v13, o23, o21));
408 }
