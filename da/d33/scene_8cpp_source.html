826  const double shiftZ = (this->buildPlateSize[1] / 2.0) - (this->buildPlateSize[1] - length[2]) / 2.0;

910 std::array<double, 6> Scene::getMeshWorldCoordinateBounds() const
911 {
912  vtkSmartPointer<vtkPoints> polyDataPoints = this->meshPolyData->GetPoints();
913  const vtkIdType numberOfPoints = polyDataPoints->GetNumberOfPoints();
914 
915  double minX = std::numeric_limits<double>::max();
916  double minY = std::numeric_limits<double>::max();
917  double minZ = std::numeric_limits<double>::max();
918 
919  double maxX = -std::numeric_limits<double>::max();
920  double maxY = -std::numeric_limits<double>::max();
921  double maxZ = -std::numeric_limits<double>::max();
922 
923  for(vtkIdType i = 0; i < numberOfPoints; i++)
924  {
925  double point[3];
926  polyDataPoints->GetPoint(i, point);
927 
928  minX = std::min(minX, point[0]);
929  minY = std::min(minY, point[1]);
930  minZ = std::min(minZ, point[2]);
931 
932  maxX = std::max(maxX, point[0]);
933  maxY = std::max(maxY, point[1]);
934  maxZ = std::max(maxZ, point[2]);
935  }
936 
937  std::array<double, 6> bounds;
938 
939  bounds[Constants::VTK_X_MIN] = minX;
940  bounds[Constants::VTK_Y_MIN] = minY;
941  bounds[Constants::VTK_Z_MIN] = minZ;
942 
943  bounds[Constants::VTK_X_MAX] = maxX;
944  bounds[Constants::VTK_Y_MAX] = maxY;
945  bounds[Constants::VTK_Z_MAX] = maxZ;
946 
947  return bounds;
948 }
