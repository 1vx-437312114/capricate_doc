112  /// @brief Compute voxel grid integer index equivalent from world coordinate.
113  ///
114  /// @param point Point in world coordinates
115  /// @param scalesize Size the mesh has been scaled to
116  /// @param numCells Voxel grid resolution
117  /// @return 3D grid index coordinate
118  ///
119  inline vtkVector3i posToCell(const vtkVector3d& point, double scalesize, int numCells[3])
120  {
121  const double xMin = 0.0;
122  const double xMax = scalesize;
123 
124  const double yMin = 0.0;
125  const double yMax = scalesize;
126 
127  const double zMin = 0.0;
128  const double zMax = scalesize;
129 
130 
131  double index[3];
132  index[0] = ( (point[0] - xMin) / (xMax - xMin) ) * numCells[0];
133  index[1] = ( (point[1] - yMin) / (yMax - yMin) ) * numCells[1];
134  index[2] = ( (point[2] - zMin) / (zMax - zMin) ) * numCells[2];
135 
136  vtkVector3i result;
137  result[0] = clampInt(static_cast<int>(index[0]), 0, numCells[0] - 1);
138  result[1] = clampInt(static_cast<int>(index[1]), 0, numCells[1] - 1);
139  result[2] = clampInt(static_cast<int>(index[2]), 0, numCells[2] - 1);
140 
141  return result;
142  }

145  /// Compute world coordinate equivalent of voxel grid integer index.
146  ///
147  /// @param index Point in world coordinates
148  /// @param scalesize Size the mesh has been scaled to
149  /// @param numCells Voxel grid resolution
150  /// @return 3D grid index coordinate
151  ///
152  inline vtkVector3d cellToPos(vtkVector3i index, double scalesize, int numCells[3])
153  {
154  const double xMin = 0.0;
155  const double xMax = scalesize;
156 
157  const double yMin = 0.0;
158  const double yMax = scalesize;
159 
160  const double zMin = 0.0;
161  const double zMax = scalesize;
162 
163 
164  double cellWidth[3];
165  cellWidth[0] = (xMax - xMin) / numCells[0];
166  cellWidth[1] = (yMax - yMin) / numCells[1];
167  cellWidth[2] = (zMax - zMin) / numCells[2];
168 
169  vtkVector3d result;
170  result[0] = index[0] * cellWidth[0] + xMin + 0.5 * cellWidth[0];
171  result[1] = index[1] * cellWidth[1] + yMin + 0.5 * cellWidth[1];
172  result[2] = index[2] * cellWidth[2] + zMin + 0.5 * cellWidth[2];
173 
174  return result;
175  }

290  /// Converts an OFF file string to a vtkPolyData object.
291  inline vtkSmartPointer<vtkPolyData> convertOFFStrToPolyData(std::string offstring)
292  {
293  std::istringstream in(offstring);
294 
295  // intialise some structures to store the file contents in
296  vtkPoints *points = vtkPoints::New();
297  vtkCellArray *polys = vtkCellArray::New();
298 
299  // -- work through the file line by line, assigning into the above structures as appropriate --
300  std::string line;
301 
302  std::getline(in, line); //throw the 1st line away, it always says "OFF"
303 
304  //Read until we get to the first line that is not a comment or a blank line.
305  //This line second states the number of vertices, faces, and edges.
306  while(std::getline(in, line))
307  {
308  //std::cout << line << std::endl;
309  if(line.size() == 0)
310  continue;
311 
312  if(line[0] == '#')
313  continue;
314 
315  //if we get to here, this is the info line
316  break;
317  }
318 
319  //std::cout << "Found info line." << std::endl;
320  //At this point, the line were are interested in is stored in 'line'
321  //We are only interested in vertices and faces.
322  std::stringstream ss;
323  ss << line;
324  unsigned int NumberOfVertices, NumberOfFaces;
325  ss >> NumberOfVertices >> NumberOfFaces;
326  //std::cout << "Vertices: " << NumberOfVertices << " Faces: " << NumberOfFaces << std::endl;
327  //Now we can start to read the vertices
328  unsigned int VertexCounter = 0;
329 
330  while(getline(in, line) && VertexCounter < NumberOfVertices)
331  {
332  if(line.size() == 0) //skip blank lines (they should only occur before the vertices start)
333  continue;
334 
335  std::stringstream ssVertex;
336  ssVertex << line;
337  float x,y,z;
338  ssVertex >> x >> y >> z;
339  points->InsertNextPoint(x, y, z);
340  //std::cout << "adding vertex: " << x << " " << y << " " << z << std::endl;
341  VertexCounter++;
342  } // (end of vertex while loop)
343 
344  unsigned int FaceCounter = 0;
345 
346  //read faces
347  do
348  {
349  std::stringstream ssFace;
350  ssFace << line;
351  unsigned int NumFaceVerts, Vert0, Vert1, Vert2;
352  ssFace >> NumFaceVerts >> Vert0 >> Vert1 >> Vert2;
353  if(NumFaceVerts != 3)
354  {
355  return nullptr;
356  }
357 
358  //std::cout << "adding triangle: " << Vert0 << " " << Vert1 << " " << Vert2 << std::endl;
359  vtkTriangle* triangle = vtkTriangle::New();
360 
361  triangle->GetPointIds()->SetId(0, Vert0);
362  triangle->GetPointIds()->SetId(1, Vert1);
363  triangle->GetPointIds()->SetId(2, Vert2);
364 
365  polys->InsertNextCell(triangle);
366 
367  triangle->Delete();
368 
369  FaceCounter++;
370 
371  } while(std::getline(in, line) && FaceCounter < NumberOfFaces);
372 
373  vtkSmartPointer<vtkPolyData> output = vtkSmartPointer<vtkPolyData>::New();
374  output->SetPoints(points);
375  output->SetPolys(polys);
376 
377  points->Delete();
378  polys->Delete();
379 
380  return output;
381  }
