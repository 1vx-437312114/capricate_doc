51 /**
52  * An efficient class for 3D triangular mesh processing.
53  * The vtkPolyData data structure wasn't designed for manipulating
54  * triangular meshes (but only to visualize them). It misses a true notion
55  * of edge with efficient accesses to adjacent vertices, adjacent faces,
56  * and more generaly the traversal of the graph of edges (like in half-edge
57  * data structure).
58  * The vtkSurfaceBase Class is an attempt to represent a low-level
59  * (because the internal representation is based on arrays and not
60  * pointers) edge-oriented 3D triangular mesh within the
61  * vtk hierarchy, and enables constant-time connectivity queries,
62  * edges labeling and easy update functionnalities.
63  */

416  vtkIdType &f2, vtkIdType &v3)
417 {
418  vtkIdType edge=this->IsEdge(v1,v2);

432

435  f2=this->Poly1->GetValue(edge);
436  }
437  v3=this->GetThirdPoint(f2,v1,v2);
