43 PM_Rational orient2D(const PM_Rational& px, const PM_Rational& py, const PM_Rational& qx, const PM_Rational& qy, const PM_Rational& rx, const PM_Rational& ry)
44 {
45  if (!PM_Rational::isUsingFiltering() && !PM_Rational::isUsingRationals())
46  {
47  return ((px - rx)*(qy - ry) - (py - ry)*(qx - rx));
48  }
49  else if (px.isOfDoubleType() && py.isOfDoubleType() && qx.isOfDoubleType() && qy.isOfDoubleType() && rx.isOfDoubleType() && ry.isOfDoubleType())
50  {
51  double pqr[6];
52  pqr[0] = px.getDVal(); pqr[1] = py.getDVal();
53  pqr[2] = qx.getDVal(); pqr[3] = qy.getDVal();
54  pqr[4] = rx.getDVal(); pqr[5] = ry.getDVal();
55  return orient2d(pqr, pqr + 2, pqr + 4);
56  }
57  else if (PM_Rational::isUsingRationals())
58  {
59  return ((px - rx)*(qy - ry) - (py - ry)*(qx - rx));
60  }
61  else
62  {
63  PM_Rational::use_rationals = true;
64  PM_Rational O = ((px - rx)*(qy - ry) - (py - ry)*(qx - rx));
65  PM_Rational::use_rationals = false;
66  return O;
67  }

269 void Point::rotate(const Point& a, const double& ang)
270 {
271  double l, q[4], m[3][3];
272  if ((l = a.length())==0.0) return;
273  l = sin(ang/2.0)/l;
274 
275  q[0] = TMESH_TO_DOUBLE(a.x)*l;
276  q[1] = TMESH_TO_DOUBLE(a.y)*l;
277  q[2] = TMESH_TO_DOUBLE(a.z)*l;
278  q[3] = cos(ang/2.0);
279 
280  m[0][0] = 1.0 - (q[1]*q[1] + q[2]*q[2])*2.0;
281  m[0][1] = (q[0] * q[1] + q[2] * q[3])*2.0;
282  m[0][2] = (q[2] * q[0] - q[1] * q[3])*2.0;
283 
284  m[1][0] = (q[0] * q[1] - q[2] * q[3])*2.0;
285  m[1][1] = 1.0 - (q[2] * q[2] + q[0] * q[0])*2.0;
286  m[1][2] = (q[1] * q[2] + q[0] * q[3])*2.0;
287 
288  m[2][0] = (q[2] * q[0] + q[1] * q[3])*2.0;
289  m[2][1] = (q[1] * q[2] - q[0] * q[3])*2.0;
290  m[2][2] = 1.0 - (q[1] * q[1] + q[0] * q[0])*2.0;
291 
292  q[0] = TMESH_TO_DOUBLE(x); q[1] = TMESH_TO_DOUBLE(y); q[2] = TMESH_TO_DOUBLE(z);
293  x = m[0][0]*q[0] + m[1][0]*q[1] + m[2][0]*q[2];
294  y = m[0][1]*q[0] + m[1][1]*q[1] + m[2][1]*q[2];
295  z = m[0][2]*q[0] + m[1][2]*q[1] + m[2][2]*q[2];
296 }

593 {

596  coord mx = MIN(s1->x, s2->x);
597  if (v1->x < mx && v2->x < mx && v3->x < mx) return false;
598  mx = MAX(s1->x, s2->x);
599  if (v1->x > mx && v2->x > mx && v3->x > mx) return false;
600  mx = MIN(s1->y, s2->y);
601  if (v1->y < mx && v2->y < mx && v3->y < mx) return false;
602  mx = MAX(s1->y, s2->y);
603  if (v1->y > mx && v2->y > mx && v3->y > mx) return false;
604  mx = MIN(s1->z, s2->z);
605  if (v1->z < mx && v2->z < mx && v3->z < mx) return false;
606  mx = MAX(s1->z, s2->z);
607  if (v1->z > mx && v2->z > mx && v3->z > mx) return false;
608 
609  o1 = s1->exactOrientation(v1, v2, v3);
610  o2 = s2->exactOrientation(v1, v2, v3);
611  if (o1 == 0 && o2 == 0)
612  {
613  if (!s1->exactSameSideOnPlane(s2, v1, v2) && !v1->exactSameSideOnPlane(v2, s1, s2)) return true;
614  if (!s1->exactSameSideOnPlane(s2, v2, v3) && !v2->exactSameSideOnPlane(v3, s1, s2)) return true;
615  if (!s1->exactSameSideOnPlane(s2, v3, v1) && !v3->exactSameSideOnPlane(v1, s1, s2)) return true;
616  if (Point::pointInInnerTriangle(s1, v1, v2, v3) && Point::pointInInnerTriangle(s2, v1, v2, v3)) return true;
617  return false;
618  }
619 
620  if ((o1>0 && o2>0) || (o1<0 && o2<0)) return false; // s1 and s2 are both above/below v1,v2,v3
621  o1 = s1->exactOrientation(s2, v1, v2);
622  o2 = s1->exactOrientation(s2, v2, v3);
623  if ((o1>0 && o2<0) || (o1<0 && o2>0)) return false;
624  o3 = s1->exactOrientation(s2, v3, v1);
625  if ((o1>0 && o3<0) || (o1<0 && o3>0)) return false;
626  if ((o2>0 && o3<0) || (o2<0 && o3>0)) return false;
627  return true;
