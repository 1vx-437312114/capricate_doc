271 // Invert the matrix. If singular return FALSE
272 
273 bool SymMatrix3x3::invert()
274 {
275  double det, pos, neg, t, out[6];
276 
277  pos = neg = 0.0;
278  t = M[0]*M[2]*M[5]; ((t >= 0.0)?(pos):(neg)) += t;
279  t = M[1]*M[4]*M[3]; ((t >= 0.0)?(pos):(neg)) += t;
280  t = M[3]*M[1]*M[4]; ((t >= 0.0)?(pos):(neg)) += t;
281  t = -M[3]*M[2]*M[3]; ((t >= 0.0)?(pos):(neg)) += t;
282  t = -M[1]*M[1]*M[5]; ((t >= 0.0)?(pos):(neg)) += t;
283  t = -M[0]*M[4]*M[4]; ((t >= 0.0)?(pos):(neg)) += t;
284  det = pos+neg;
285 
286  t = det/(pos-neg);
287  if (FABS(t) >= 1.0e-15)
288  {
289  out[0] = (M[2] * M[5] - M[4] * M[4]) /det;
290  out[1] = -(M[1] * M[5] - M[4] * M[3]) /det;
291  out[2] = (M[0] * M[5] - M[3] * M[3]) /det;
292  out[3] = (M[1] * M[4] - M[2] * M[3]) /det;
293  out[4] = -(M[0] * M[4] - M[1] * M[3]) /det;
294  out[5] = (M[0] * M[2] - M[1] * M[1]) /det;
295  M[0] = out[0]; M[1] = out[1]; M[2] = out[2];
296  M[3] = out[3]; M[4] = out[4]; M[5] = out[5];
297  return 1;
298  }
299 
300  return 0;
301 }

304 // Compute eigenvalues and eigenvectors of the matrix (JACOBI method).
305 // The calling function is responsible of verifying that the matrix
306 // is diagonalizable.
307 //
308 // This routine was inspired from a software to estimate curvature
309 // tensors developed at INRIA. Visit the following link for details:
310 // http://www-sop.inria.fr/geometrica/team/Pierre.Alliez/demos/curvature/
311 //
312 // This version has been slightly optimized.
313 //
314 
315 void SymMatrix3x3::diagonalize(double *eigen_val, double *eigen_vec) const
316 {
317  static const double EPS = 0.00001;
318  static const double cos_pi_4 = 0.70710678;
319  static const int MAX_ITER = 100;
320  double a_P[6], v_P[9];
321  double *a = a_P, *v = v_P;
322  double a_norm,a_normEPS,thr,thr_nn;
323  int nb_iter = 0;
324  int i,j,k,ij,jj,ik,l,m,lm,mq,lq,ll,mm,imv,im,iq,ilv,il;
325  int index_P[3];
326  int *index = index_P;
327  double a_ij,a_lm,a_ll,a_mm,a_im,a_il,a_lm_2,v_ilv,v_imv,x;
328  double sinx,sinx_2,cosx,cosx_2,sincos,delta;
329 
330  a[0] = M[0]; a[1] = M[1]; a[2] = M[2];
331  a[3] = M[3]; a[4] = M[4]; a[5] = M[5];
332  a--;
333 
334  // Step 2 : Init diagonalization matrix as the unit matrix
335 
336  for (ij=0, i=0; i<3; i++) for (j=0; j<3; j++) v[ij++] = (i==j)?(1.0):(0.0);
337  v--;
338 
339  // Step 3 : compute the weight of the non diagonal terms
340 
341  a_norm = 0.0;
342  for (i=1, ij=1; i<=3; i++) for (j=1; j<=i; j++, ij++) if( i!=j ) {a_ij = a[ij]; a_norm += a_ij*a_ij;}
343 
344  if( a_norm != 0.0 )
345  {
346  a_normEPS = a_norm*EPS;
347  thr = a_norm ;
348 
349  // Step 4 : rotations
350  while (thr > a_normEPS && nb_iter < MAX_ITER)
351  {
352  nb_iter++;
353  thr_nn = thr / 6;
354 
355  for (l=1; l<3; l++)
356  for (m=l+1; m<=3; m++)
357  {
358  // compute sinx and cosx
359 
360  lq = (l*l-l)/2; mq = (m*m-m)/2;
361  lm = l+mq; a_lm = a[lm];
362  a_lm_2 = a_lm*a_lm;
363 
364  if( a_lm_2 < thr_nn ) continue;
365 
366  ll = l+lq; mm = m+mq;
367  a_ll = a[ll]; a_mm = a[mm];
368  delta = a_ll - a_mm;
369 
370  if (delta==0.0) {sinx = -cos_pi_4; cosx = cos_pi_4;}
371  else {x = -atan( (a_lm+a_lm) / delta )/2.0; sinx=sin(x); cosx=cos(x);}
372 
373  sinx_2 = sinx*sinx;
374  cosx_2 = cosx*cosx;
375  sincos = sinx*cosx;
376 
377  // rotate L and M columns
378 
379  ilv = 3*(l-1); imv = 3*(m-1);
380 
381  for( i=1; i<=3;i++ )
382  {
383  if( (i!=l) && (i!=m) )
384  {
385  iq = (i*i-i)/2;
386  im = (i<m)?(i+mq):(m+iq);
387  a_im = a[im];
388  il = (i<l)?(i+lq):(l+iq);
389  a_il = a[il];
390  a[il] = a_il*cosx - a_im*sinx;
391  a[im] = a_il*sinx + a_im*cosx;
392  }
393 
394  ilv++; imv++;
395  v_ilv = v[ilv]; v_imv = v[imv];
396  v[ilv] = cosx*v_ilv - sinx*v_imv;
397  v[imv] = sinx*v_ilv + cosx*v_imv;
398  }
399 
400  x = a_lm*sincos; x+=x;
401 
402  a[ll] = a_ll*cosx_2 + a_mm*sinx_2 - x;
403  a[mm] = a_ll*sinx_2 + a_mm*cosx_2 + x;
404  a[lm] = 0.0;
405  thr = FABS( thr - a_lm_2 );
406  }
407  }
408  }
409 
410  // Step 5: index conversion and copy eigen values
411 
412  // back from Fortran to C++
413  a++;
414 
415  for( i=0; i<3; i++ ) { k = i + (i*(i+1))/2; eigen_val[i] = a[k];}
416 
417  // Step 6: sort the eigen values and eigen vectors
418 
419  for( i=0; i<3; i++ ) index[i] = i;
420 
421  for( i=0; i<(3-1); i++ ) {
422  x = eigen_val[i];
423  k = i;
424 
425  for( j=i+1; j<3; j++ ) if( x < eigen_val[j] ) { k = j; x = eigen_val[j];}
426 
427  eigen_val[k] = eigen_val[i];
428  eigen_val[i] = x;
429 
430  jj = index[k];
431  index[k] = index[i];
432  index[i] = jj;
433  }
434 
435  // Step 7: save the eigen vectors
436 
437  v++; // back from Fortran to to C++
438 
439  ij = 0;
440  for( k=0; k<3; k++ ) for(ik = index[k]*3, i=0; i<3; i++ ) eigen_vec[ij++] = v[ik++];
441 }

446 void SymMatrix3x3::getEigenvalues(double *L1, double *L2, double *L3) const
447 {
448  double a11 = M[0], a12 = M[1], a22 = M[2], a13 = M[3], a23 = M[4], a33 = M[5];
449  double c0 = (a11*a22*a33)+(2.0*a12*a13*a23)-(a11*a23*a23)-(a22*a13*a13)-(a33*a12*a12); // Mat. Determinant
450  double c1 = a11*a22-a12*a12+a11*a33-a13*a13+a22*a33-a23*a23;
451  double c2 = a11+a22+a33;
452  double a = (3*c1-c2*c2)/3.0;
453  double b = (9*c1*c2-2*c2*c2*c2-27*c0)/27.0;
454  double Q = ((b*b)/4.0)+((a*a*a)/27.0);
455 
456  if (Q>1.0e-12) {*L1 = *L2 = *L3 = a11; return;} // Evecs = (1,0,0), (0,1,0), (0,0,1)
457 
458  double l1, l2, l3; // Eigenvalues to be computed
459 
460  if (Q>=0) {double p=(b>0)?(pow(b/2, 1.0/3.0)):(0); l1=l2=((c2/3.0)+p); l3=((c2/3.0)-2.0*p);}
461  else
462  {
463  double t = atan2(sqrt(-Q), -b/2.0)/3.0, r = pow(((b*b)/4.0)-Q, 1.0/6.0);
464  double cos_t = cos(t), sin_t = sin(t);
465  const double sq3 = sqrt(3.0);
466  l1 = l2 = l3 = (c2/3.0);
467  l1 += (2*r*cos_t);
468  l2 -= r*(cos_t+sq3*sin_t);
469  l3 -= r*(cos_t-sq3*sin_t);
470  }
471 
472  if (l1<=l2 && l1<=l3) {*L1=l1; *L2=(l2<l3)?(l2):(l3); *L3=(l2<l3)?(l3):(l2);}
473  else if (l2<=l1 && l2<=l3) {*L1=l2; *L2=(l1<l3)?(l1):(l3); *L3=(l1<l3)?(l3):(l1);}
474  else {*L1=l3; *L2=(l1<l2)?(l1):(l2); *L3=(l1<l2)?(l2):(l1);}
475 }

480 double SymMatrix3x3::getMinEigenvector(double *x, double *y, double *z) const
481 {
482  double a11 = M[0], a12 = M[1], a22 = M[2], a13 = M[3], a23 = M[4], a33 = M[5];
483  double l, l1, l2, l3, c0, c1, c2;
484  getEigenvalues(&l, &l2, &l3);
485 
486  if (l==l3 && l==l2) {*x = 1; *y = *z = 0; return l;}
487 
488  a11-=l; a22-=l; a33-=l;
489  double u11 = a22*a33-a23*a23, u12 = a13*a23-a12*a33, u13 = a12*a23-a13*a22;
490  double u22 = a11*a33-a13*a13, u23 = a12*a13-a23*a11, u33 = a11*a22-a12*a12;
491  l1 = u11*u11+u12*u12+u13*u13;
492  l2 = u12*u12+u22*u22+u23*u23;
493  l3 = u13*u13+u23*u23+u33*u33;
494 
495  if (l1>=l2 && l1>=l3) {c0=u11; c1=u12; c2=u13; l=l1;}
496  else if (l2>=l1 && l2>=l3) {c0=u12; c1=u22; c2=u23; l=l2;}
497  else {c0=u13; c1=u23; c2=u33; l=l3;}
498 
499  l1 = sqrt(l); *x = c0/l1; *y = c1/l1; *z = c2/l1;
500 
501  return l;
502 }

625 bool SymMatrix4x4::getMinimizer(coord *a, coord *b, coord *c) const
626 {
627  coord det, pos, neg, t;
628 
629  pos = neg = 0.0;
630  t = a2*b2*c2; ((t >= 0.0)?(pos):(neg)) += t;
631  t = ab*bc*ac; ((t >= 0.0)?(pos):(neg)) += t;
632  t = ac*ab*bc; ((t >= 0.0)?(pos):(neg)) += t;
633  t = -ac*b2*ac; ((t >= 0.0)?(pos):(neg)) += t;
634  t = -ab*ab*c2; ((t >= 0.0)?(pos):(neg)) += t;
635  t = -a2*bc*bc; ((t >= 0.0)?(pos):(neg)) += t;
636  det = pos+neg;
637 
638  if (pos == neg) return 0;
639 
640  t = det/(pos-neg);
641  if (
642 #ifdef USE_HYBRID_KERNEL
643  (coord::isUsingRationals() && t!=0) ||
644 #endif
645  (FABS(t) >= 1.0e-15))
646  {
647  *a = -(ad*( (b2*c2 - bc*bc)) + bd*(-(ab*c2 - bc*ac)) + cd*( (ab*bc - b2*ac)))/det;
648  *b = -(ad*(-(ab*c2 - ac*bc)) + bd*( (a2*c2 - ac*ac)) + cd*(-(a2*bc - ab*ac)))/det;
649  *c = -(ad*( (ab*bc - ac*b2)) + bd*(-(a2*bc - ac*ab)) + cd*( (a2*b2 - ab*ab)))/det;
650  return 1;
651  }
652 
653  return 0;
654 }

657 // Invert a symmetric 4x4 matrix using L*D*L^T decomposition.
658 // The calling function is responsible of verifying that the matrix
659 // is positive definite.
660 //
661 
662 bool SymMatrix4x4::invert()
663 {
664  if (a2 <= 0) return false;
665 
666  coord d00 = coord(1.0) / a2;
667  coord L10 = ab;
668  coord l10 = ab*d00;
669  coord L20 = ac;
670  coord l20 = ac*d00;
671  coord L30 = ad;
672  coord l30 = ad*d00;
673  coord d11 = b2-(L10*l10);
674 
675  if (d11 <= 0) return false; else d11 = coord(1.0)/d11;
676 
677  coord L21 = (bc-(L10*l20));
678  coord l21 = L21*d11;
679  coord L31 = (bd-(L10*l30));
680  coord l31 = L31*d11;
681  coord d22 = c2-(L20*l20)-(L21*l21);
682 
683  if (d22 <= 0) return false; else d22 = coord(1.0) / d22;
684 
685  coord L32 = (cd-(L20*l30)-(L21*l31));
686  coord l32 = L32*d22;
687  coord d33 = d2-(L30*l30)-(L31*l31)-(L32*l32);
688 
689  if (d33 <= 0) return false; else d33 = coord(1.0) / d33;
690 
691  L20 = l10*l21-l20;
692  L31 = l21*l32-l31;
693  L30 = l32*l20-L31*l10-l30;
694 
695  a2 = d00+(-l10)*((-l10)*d11)+L20*(L20*d22)+L30*(L30*d33);
696  ab = ((-l10)*d11)+L20*((-l21)*d22)+L30*(L31*d33);
697  b2 = d11+(-l21)*((-l21)*d22)+L31*(L31*d33);
698  ac = (L20*d22)+L30*((-l32)*d33);
699  bc = ((-l21)*d22)+L31*((-l32)*d33);
700  c2 = d22+(-l32)*((-l32)*d33);
701  ad = (L30*d33);
702  bd = (L31*d33);
703  cd = ((-l32)*d33);
704  d2 = d33;
705 
706  return true;
707 }
