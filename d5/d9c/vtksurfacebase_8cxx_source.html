186 bool vtkSurfaceBase::IsVertexManifold( const vtkIdType& iV )
187 {
188  vtkIdType v1,v2;
189  vtkIdType FirstEdge,FirstVertex;
190  vtkIdType f1,f2,f3;
191  vtkIdType *Edges,NumberOfRemainingEdges;
192 
193  this->GetVertexNeighbourEdges(iV,NumberOfRemainingEdges,Edges);
194 
195  // if there is only one edge, the vertex is non manifold
196  if (NumberOfRemainingEdges<2)
197  return (false);
198 
199  // detect non-manifold edges
200  for (vtkIdType i=0;i<NumberOfRemainingEdges;i++)
201  {
202  if (!this->IsEdgeManifold(Edges[i]))
203  return (false);
204  }
205 
206  FirstEdge=this->GetFirstEdge(iV);
207  this->GetEdgeVertices(FirstEdge,FirstVertex,v1);
208  if (FirstVertex==iV)
209  FirstVertex=v1;
210  v1=FirstVertex;
211 
212  NumberOfRemainingEdges--;
213  this->GetEdgeFaces(FirstEdge,f1,f2);
214 
215  // turn in the first direction
216  v2=this->GetThirdPoint(f1,iV,FirstVertex);
217  do
218  {
219  if (--NumberOfRemainingEdges==0)
220  return (true);
221  Conquer(f1,iV,v2,f3,v1);
222  f1=f3;
223  v2=v1;
224  } while ((f1>=0)&&(v2!=FirstVertex));
225 
226  // if we turned around back to the first vertex, or there is only one direction (f2==-1) return false
227  if ((f2<0)||(v2==FirstVertex))
228  return (false);
229 
230  // turn in the second direction
231  v1=FirstVertex;
232  v2=this->GetThirdPoint(f2,iV,FirstVertex);
233  do
234  {
235  if (--NumberOfRemainingEdges==0)
236  return (true);
237  Conquer(f2,iV,v2,f3,v1);
238  f2=f3;
239  v2=v1;
240 
241  } while ((f2>=0)&&(v2!=FirstVertex));
242  // all adjacent faces have been visited, but there are edges remaining, so return false
243  return (false);
244 }

333 void vtkSurfaceBase::ConquerOrientationFromFace(vtkIdType Face)
334 {
335  std::queue <vtkIdType> EdgesQueue;
336  vtkIdType v1,v2,v3,v4,f1,f2,Edge,Edge2;
337  vtkIdType Visited1,Visited2;
338  vtkIdType NumberOfPoints1,NumberOfPoints2,*Face1,*Face2;
339  vtkIdType j;
340  vtkIdType FlipFace;
341 
342  this->GetCellPoints(Face,NumberOfPoints1,Face1);
343  for (j=0;j<NumberOfPoints1;j++)
344  {
345  Edge=this->IsEdge(Face1[j],Face1[(j+1)%NumberOfPoints1]);
346  if (this->IsEdgeManifold(Edge))
347  EdgesQueue.push(Edge);
348  }
349 
350  int Index1,Index2;
351 
352  while (EdgesQueue.size())
353  {
354  // pop an edge from the queue
355  Edge=EdgesQueue.front();
356  EdgesQueue.pop();
357  this->GetEdgeFaces(Edge,f1,f2);
358  if (f2>=0)
359  {
360  Visited1=this->VisitedPolygons->GetValue(f1);
361  Visited2=this->VisitedPolygons->GetValue(f2);
362  if ((Visited1!=Visited2))
363  {
364  // if one adjacent face was not visited yet
365  if (Visited2==1)
366  {
367  v1=f1;
368  f1=f2;
369  f2=v1;
370  Visited1=Visited2;
371  }
372  // f1 was wisited, not f2
373  FlipFace=0;
374  this->GetCellPoints(f1,NumberOfPoints1,Face1);
375  this->GetCellPoints(f2,NumberOfPoints2,Face2);
376  this->GetEdgeVertices(Edge,v1,v2);
377  Index1=FindVertexIndex(Face1,v1,NumberOfPoints1);
378  Index2=FindVertexIndex(Face2,v1,NumberOfPoints2);
379 
380  v3=Face1[(Index1+1)%NumberOfPoints1];
381  v4=Face2[(Index2+1)%NumberOfPoints2];
382  if (((v3==v2)&&(v4==v2))||((v3!=v2)&&(v4!=v2)))
383  {
384  // flip f2
385  for (j=0;j<NumberOfPoints2/2;j++)
386  {
387  v1=Face2[j];
388  Face2[j]=Face2[NumberOfPoints2-1-j];
389  Face2[NumberOfPoints2-1-j]=v1;
390  }
391  }
392 
393  this->VisitedPolygons->SetValue(f2,1);
394  // add adjacent edges of the face to the queue, except Edge
395  for (j=0;j<NumberOfPoints2;j++)
396  {
397  Edge2=this->IsEdge(Face2[j],Face2[(j+1)%NumberOfPoints2]);
398  if ((Edge2!=Edge)&&(this->IsEdgeManifold(Edge2)==1))
399  EdgesQueue.push(Edge2);
400  }
401  }
402  }
403  }

603 void vtkSurfaceBase::MergeVertices(vtkIdType v1, vtkIdType v2)
604 {

606  vtkIdList *List2=vtkIdList::New();

609

613  this->GetEdgeFaces(Edge,List);
614  for (int i=0;i<List->GetNumberOfIds();i++)
615  this->DeleteFace(List->GetId(i));
616  }
617

619  this->GetVertexNeighbourFaces(v2,List);
620  for (vtkIdType i=0;i<List->GetNumberOfIds();i++)

631  {
