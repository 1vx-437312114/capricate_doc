50 #define FABS(a) (((a)>=0.0)?(a):(-(a)))
51 #define FTST(a,b,x,y) _bvr=x-a; y=b-_bvr
52 #define FTS(a,b,x,y) x=(double)(a+b); FTST(a,b,x,y)
53 #define TST(a,b,x,y) _bvr=(double)(x-a); _avr=x-_bvr; _brn=b-_bvr; _arn=a-_avr; y=_arn+_brn
54 #define TWS(a,b,x,y) x=(double)(a+b); TST(a,b,x,y)
55 #define TDT(a,b,x,y) _bvr=(double)(a-x); _avr=x+_bvr; _brn=_bvr-b; _arn=a-_avr; y=_arn+_brn
56 #define TWD(a,b,x,y) x=(double)(a-b); TDT(a,b,x,y)
57 #define SPLT(a,ahi,alo) c=(double)(_spl*a); abig=(double)(c-a); ahi=c-abig; alo=a-ahi
58 #define TPT(a,b,x,y) SPLT(a,ahi,alo); SPLT(b,bhi,blo); err1=x-(ahi*bhi); err2=err1-(alo*bhi); err3=err2-(ahi*blo); y=(alo*blo)-err3
59 #define TWP(a,b,x,y) x=(double)(a*b); TPT(a,b,x,y)
60 #define TPP(a,b,bhi,blo,x,y) x=(double)(a*b); SPLT(a,ahi,alo); err1=x-(ahi*bhi); err2=err1-(alo*bhi); err3=err2-(ahi*blo); y=(alo*blo)-err3
61 #define TOD(a1,a0,b,x2,x1,x0) TWD(a0,b,_i,x0); TWS(a1,_i,x2,x1)
62 #define TTD(a1,a0,b1,b0,x3,x2,x1,x0) TOD(a1,a0,b0,_j,_0,x0); TOD(_j,_0,b1,x3,x2,x1)

68 int _fesze(int elen, double *e, int flen, double *f, double *h)
69 {
70  double Q, Qnew, hh, _bvr, _avr, _brn, _arn, enow, fnow;
71  int eindex, findex, hindex;
72 
73  enow = e[0];
74  fnow = f[0];
75  eindex = findex = 0;
76  if ((fnow > enow) == (fnow > -enow)) {
77  Q = enow;
78  enow = e[++eindex];
79  } else {
80  Q = fnow;
81  fnow = f[++findex];
82  }
83  hindex = 0;
84  if ((eindex < elen) && (findex < flen)) {
85  if ((fnow > enow) == (fnow > -enow)) {
86  FTS(enow, Q, Qnew, hh);
87  enow = e[++eindex];
88  } else {
89  FTS(fnow, Q, Qnew, hh);
90  fnow = f[++findex];
91  }
92  Q = Qnew;
93  if (hh != 0.0) {
94  h[hindex++] = hh;
95  }
96  while ((eindex < elen) && (findex < flen)) {
97  if ((fnow > enow) == (fnow > -enow)) {
98  TWS(Q, enow, Qnew, hh);
99  enow = e[++eindex];
100  } else {
101  TWS(Q, fnow, Qnew, hh);
102  fnow = f[++findex];
103  }
104  Q = Qnew;
105  if (hh != 0.0) {
106  h[hindex++] = hh;
107  }
108  }
109  }
110  while (eindex < elen) {
111  TWS(Q, enow, Qnew, hh);
112  enow = e[++eindex];
113  Q = Qnew;
114  if (hh != 0.0) {
115  h[hindex++] = hh;
116  }
117  }
118  while (findex < flen) {
119  TWS(Q, fnow, Qnew, hh);
120  fnow = f[++findex];
121  Q = Qnew;
122  if (hh != 0.0) {
123  h[hindex++] = hh;
124  }
125  }
126  if ((Q != 0.0) || (hindex == 0)) {
127  h[hindex++] = Q;
128  }
129  return hindex;
130 }

132 int _seze(int elen, double *e, double b, double *h)
133 {
134  double Q, sum, hh, product1, product0, enow, _bvr, _avr, _brn, _arn, c;
135  double abig, ahi, alo, bhi, blo, err1, err2, err3;
136  int eindex, hindex;
137 
138  SPLT(b, bhi, blo);
139  TPP(e[0], b, bhi, blo, Q, hh);
140  hindex = 0;
141  if (hh != 0) {
142  h[hindex++] = hh;
143  }
144  for (eindex = 1; eindex < elen; eindex++) {
145  enow = e[eindex];
146  TPP(enow, b, bhi, blo, product1, product0);
147  TWS(Q, product0, sum, hh);
148  if (hh != 0) {
149  h[hindex++] = hh;
150  }
151  FTS(product1, sum, Q, hh);
152  if (hh != 0) {
153  h[hindex++] = hh;
154  }
155  }
156  if ((Q != 0.0) || (hindex == 0)) {
157  h[hindex++] = Q;
158  }
159  return hindex;
160 }

164  int eindex;
165  double Q = e[0];
166  for (eindex = 1; eindex < elen; eindex++) Q += e[eindex];

171 double _adaptive2dorientation(double *pa, double *pb, double *pc, double detsum)
172 {
173  double acx, acy, bcx, bcy,acxtail, acytail, bcxtail, bcytail, detleft, detright;
174  double detlefttail, detrighttail, det, errbound, B[4], C1[8], C2[12], D[16];
175  double B3, u[4], u3, s1, t1, s0, t0, _bvr, _avr, _brn, _arn, c;
176  double abig, ahi, alo, bhi, blo, err1, err2, err3, _i, _j, _0;
177  int C1length, C2length, Dlength;
178 
179  acx = (double) (pa[0] - pc[0]);
180  bcx = (double) (pb[0] - pc[0]);
181  acy = (double) (pa[1] - pc[1]);
182  bcy = (double) (pb[1] - pc[1]);
183 
184  TWP(acx, bcy, detleft, detlefttail);
185  TWP(acy, bcx, detright, detrighttail);
186 
187  TTD(detleft, detlefttail, detright, detrighttail,
188  B3, B[2], B[1], B[0]);
189  B[3] = B3;
190 
191  det = _estm(4, B);
192  errbound = _ccwebB * detsum;
193  if ((det >= errbound) || (-det >= errbound)) {
194  return det;
195  }
196 
197  TDT(pa[0], pc[0], acx, acxtail);
198  TDT(pb[0], pc[0], bcx, bcxtail);
199  TDT(pa[1], pc[1], acy, acytail);
200  TDT(pb[1], pc[1], bcy, bcytail);
201 
202  if ((acxtail == 0.0) && (acytail == 0.0)
203  && (bcxtail == 0.0) && (bcytail == 0.0)) {
204  return det;
205  }
206 
207  errbound = _ccwebC * detsum + _reb * FABS(det);
208  det += (acx * bcytail + bcy * acxtail)
209  - (acy * bcxtail + bcx * acytail);
210  if ((det >= errbound) || (-det >= errbound)) {
211  return det;
212  }
213 
214  TWP(acxtail, bcy, s1, s0);
215  TWP(acytail, bcx, t1, t0);
216  TTD(s1, s0, t1, t0, u3, u[2], u[1], u[0]);
217  u[3] = u3;
218  C1length = _fesze(4, B, 4, u, C1);
219 
220  TWP(acx, bcytail, s1, s0);
221  TWP(acy, bcxtail, t1, t0);
222  TTD(s1, s0, t1, t0, u3, u[2], u[1], u[0]);
223  u[3] = u3;
224  C2length = _fesze(C1length, C1, 4, u, C2);
225 
226  TWP(acxtail, bcytail, s1, s0);
227  TWP(acytail, bcxtail, t1, t0);
228  TTD(s1, s0, t1, t0, u3, u[2], u[1], u[0]);
229  u[3] = u3;
230  Dlength = _fesze(C2length, C2, 4, u, D);
231 
232  return(D[Dlength - 1]);
233 }
