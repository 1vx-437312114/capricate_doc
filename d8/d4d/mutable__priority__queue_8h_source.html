29 /*
30  A priority queue in which the priorities of entries may be
31  updated in lg(n) time.
32 
33  Simply two maps glued together, one mapping keys (priorities)
34  to values and the other mapping values to keys. Both keys
35  and values must meet the requirements for storage in a
36  std::map, i.e. be less than comparable. Typically not a
37  problem for priorities, but can be a pain if Values are stored
38  by value in the structures. Not a problem if pointers to
39  values are stored.
40 
41  insert, erase and update operations are all logarithmic, but
42  approximately 2x slower than std::map since two std::maps
43  are updated in each case.
44 */
45 template< typename _Key, typename _Value, typename _KeyComp=std::greater<_Key>, typename _ValueComp=std::less<_Value> >
46 class mutable_priority_queue {
47 public:
48  // iterator type for the key2val map, allows in-order traversal of the queue
49  typedef typename std::multimap<_Key,_Value,_KeyComp>::const_iterator iterator;

130  // adds a value key pair to the queue.
131  // an attempt first has to be made to erase
132  // the item, since it could exist with another
133  // key and the queue does not allow multiple
134  // keys per value
135  inline void insert( _Value val, _Key key ){
136  erase( val );
137  key2Val.insert( std::pair<_Key,_Value>( key, val ) );
138  val2Key.insert( std::pair<_Value,_Key>( val, key ) );
139  }

153 private:

155  std::map< _Value, _Key, _ValueComp > val2Key;
156 
157  inline void _erase( typename std::multimap<_Key,_Value,_KeyComp>::iterator it ){
158  val2Key.erase( it->second );
159  key2Val.erase( it );
160  }

162  inline typename std::multimap<_Key,_Value,_KeyComp>::iterator _find( _Key k, _Value v ){
163  typename std::multimap<_Key,_Value,_KeyComp>::iterator it = key2Val.lower_bound(k);
164  typename std::multimap<_Key,_Value,_KeyComp>::iterator hi = key2Val.upper_bound(k);
165  while( it != hi ){
166  if( it->second == v )
167  return it;
168  it++;
169  }
170  return key2Val.end();
171  }
