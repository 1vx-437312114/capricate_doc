138 int Basic_TMesh::selectIntersectingTriangles(UINT16 tris_per_cell, bool justproper)
139 {
140  Triangle *t;
141  Vertex *v;
142  Node *n;
143  bool isSelection=0;
144  List *selT = new List, *selV = new List;
145 
146  TMesh::begin_progress();
147  TMesh::report_progress(NULL);
148 
149  FOREACHTRIANGLE(t, n) if (IS_VISITED(t))
150  {
151  isSelection=1;
152  selT->appendTail(t);
153  v=t->v1(); if (!IS_BIT(v,5)) {MARK_BIT(v,5); selV->appendTail(v);}
154  v=t->v2(); if (!IS_BIT(v,5)) {MARK_BIT(v,5); selV->appendTail(v);}
155  v=t->v3(); if (!IS_BIT(v,5)) {MARK_BIT(v,5); selV->appendTail(v);}
156  }
157  TMesh::report_progress(NULL);
158 
159  if (!isSelection) {delete(selT); delete(selV); selT=&T; selV=&V;}
160 
161  di_cell *c2, *c = new di_cell(this, !isSelection);
162  List cells, todo(c);
163  int i=0;
164 
165  while ((c = (di_cell *)todo.popHead()) != NULL)
166  {
167  if (i>DI_MAX_NUMBER_OF_CELLS || c->triangles.numels() <= tris_per_cell) cells.appendHead(c);
168  else
169  {
170  if (!(i % 1000)) TMesh::report_progress(NULL);
171  i++;
172  c2 = c->fork();
173  todo.appendTail(c);
174  todo.appendTail(c2);
175  }
176  }
177 
178  // Deselect everything and select only intersecting triangles
179  deselectTriangles();
180  FOREACHTRIANGLE(t, n) t->info = NULL;
181  i=0; FOREACHNODE(cells, n)
182  {
183  (((di_cell *)n->data)->selectIntersections(justproper));
184  if (!(i % 100)) TMesh::report_progress("%d %% done ", ((i)* 100) / cells.numels());
185  i++;
186  }
187  TMesh::end_progress();
188 
189  // Dispose memory allocated for cells
190  FOREACHVTTRIANGLE(selT, t, n) { if (t->info!=NULL) delete((List *)t->info); t->info = NULL; }
191  while (cells.numels()) delete((di_cell *)cells.popHead());
192 
193  // Count selected triangles for final report and delete stored normals
194  int its=0;
195  FOREACHVTTRIANGLE(selT, t, n) { if (IS_VISITED(t)) its++;}
196 
197  if (its) TMesh::info("%d intersecting triangles have been selected.\n",its);
198  else TMesh::info("No intersections detected.\n");
199 
200  FOREACHVVVERTEX(selV, v, n) UNMARK_BIT(v,5);
201  if (isSelection) {delete(selT); delete(selV);}
202 
203  return its;
204 }

257 bool Basic_TMesh::safeCoordBackApproximation()
258 {
259  Node *n;
260  Vertex *v;
261 
262  deselectTriangles();
263 
264  FOREACHVERTEX(v, n)
265  {
266  jitterCoordinate(v->x, 0);
267  jitterCoordinate(v->y, 0);
268  jitterCoordinate(v->z, 0);
269  }
270 
271  Edge *e;
272  Vertex *ov1, *ov2;
273 
274  int pnos = 0, nos;
275  nos = 0; FOREACHEDGE(e, n) if (e->overlaps()) nos++;
276 
277  do
278  {
279  pnos = nos;
280  FOREACHEDGE(e, n) if (e->overlaps())
281  {
282  ov1 = e->t1->oppositeVertex(e);
283  ov2 = e->t2->oppositeVertex(e);
284  v = (Point::squaredTriangleArea3D(e->v1, e->v2, ov1) < Point::squaredTriangleArea3D(e->v1, e->v2, ov2)) ? (ov1) : (ov2);
285  for (int a = -1; a <= 1; a++) for (int b = -1; b <= 1; b++) for (int c = -1; c <= 1; c++)
286  {
287  jitterCoordinate(v->x, a); jitterCoordinate(v->y, b); jitterCoordinate(v->z, c);
288  if (e->overlaps())
289  {
290  jitterCoordinate(v->x, -a); jitterCoordinate(v->y, -b); jitterCoordinate(v->z, -c);
291  } else a = b = c = 2;
292  }
293  }
294  nos = 0; FOREACHEDGE(e, n) if (e->overlaps()) nos++;
295  } while (nos < pnos);

297 // if (nos) TMesh::warning("%d overlaps could not be removed.\n", nos);
298  return (nos == 0);

308 bool remints_appendCubeToList(Triangle *t0, List& l)
309 {
310  if (!IS_VISITED(t0) || IS_BIT(t0, 6)) return false;
311 
312  Triangle *t, *s;
313  Vertex *v;
314  List triList(t0);
315  MARK_BIT(t0,6);
316  coord minx = DBL_MAX, maxx = -DBL_MAX, miny = DBL_MAX, maxy = -DBL_MAX, minz = DBL_MAX, maxz = -DBL_MAX;
317 
318  while(triList.numels())
319  {
320  t = (Triangle *)triList.popHead();
321  v = t->v1();
322  minx=MIN(minx,v->x); miny=MIN(miny,v->y); minz=MIN(minz,v->z);
323  maxx=MAX(maxx,v->x); maxy=MAX(maxy,v->y); maxz=MAX(maxz,v->z);
324  v = t->v2();
325  minx=MIN(minx,v->x); miny=MIN(miny,v->y); minz=MIN(minz,v->z);
326  maxx=MAX(maxx,v->x); maxy=MAX(maxy,v->y); maxz=MAX(maxz,v->z);
327  v = t->v3();
328  minx=MIN(minx,v->x); miny=MIN(miny,v->y); minz=MIN(minz,v->z);
329  maxx=MAX(maxx,v->x); maxy=MAX(maxy,v->y); maxz=MAX(maxz,v->z);
330  if ((s = t->t1()) != NULL && !IS_BIT(s, 6) && IS_VISITED(s)) {triList.appendHead(s); MARK_BIT(s,6);}
331  if ((s = t->t2()) != NULL && !IS_BIT(s, 6) && IS_VISITED(s)) {triList.appendHead(s); MARK_BIT(s,6);}
332  if ((s = t->t3()) != NULL && !IS_BIT(s, 6) && IS_VISITED(s)) {triList.appendHead(s); MARK_BIT(s,6);}
333  }
334 
335  l.appendTail(new Point(minx, miny, minz));
336  l.appendTail(new Point(maxx, maxy, maxz));
337  return true;
338 }
