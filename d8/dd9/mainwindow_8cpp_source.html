740 std::shared_ptr<Voxelizer> MainWindow::getVoxelizer()
741 {
742  std::shared_ptr<Voxelizer> voxelizer;
743  int selection = this->preferencesDialog->getVoxelizer_comboBox();
744 
745  if(selection == 0)
746  {
747  voxelizer = std::make_shared<SinglePassVoxelizer>();
748  }
749  else if(selection == 1)
750  {
751  voxelizer = std::make_shared<MultiPassVoxelizer>();
752  }
753  else if(selection == 2)
754  {
755  voxelizer = std::make_shared<RobustVoxelizer>();
756  Voxelizer* v = voxelizer.get();
757  dynamic_cast<RobustVoxelizer*>(v)->setSampleSingleDirection();
758  }
759  else if(selection == 3)
760  {
761  voxelizer = std::make_shared<RobustVoxelizer>();
762  Voxelizer* v = voxelizer.get();
763  dynamic_cast<RobustVoxelizer*>(v)->setSampleSixNeighborhood();
764  }
765  else if(selection == 4)
766  {
767  voxelizer = std::make_shared<RobustVoxelizer>();
768  Voxelizer* v = voxelizer.get();
769  dynamic_cast<RobustVoxelizer*>(v)->setSampleEighteenNeighborhood();
770  }
771  else if(selection == 5)
772  {
773  voxelizer = std::make_shared<RobustVoxelizer>();
774  Voxelizer* v = voxelizer.get();
775  dynamic_cast<RobustVoxelizer*>(v)->setSampleTwentysixNeighborhood();
776  }
777  else if(selection == 6)
778  {
779  voxelizer = std::make_shared<RobustVoxelizer>();
780  Voxelizer* v = voxelizer.get();
781  dynamic_cast<RobustVoxelizer*>(v)->setSampleUnitSphere();
782  }
783  else if(selection == 7)
784  {
785  voxelizer = std::make_shared<OctreeVoxelizer>();
786  }
787  else
788  {
789  voxelizer = std::make_shared<SinglePassVoxelizer>();
790  }
791 
792 
793  return voxelizer;
794 }

797 void MainWindow::loadMesh(QString fileName, bool scaleMesh)
798 {
799  //////////////////////////////////////////////////////////////////////////////////////////////////
800  // read mesh file
801  //////////////////////////////////////////////////////////////////////////////////////////////////
802  QByteArray ba = fileName.toLatin1();
803  const char *fileName_c_str = ba.data();
804 
805  vtkSmartPointer<vtkAbstractPolyDataReader> reader;
806 
807  std::string extension = vtksys::SystemTools::GetFilenameExtension(std::string(fileName_c_str));
808  boost::to_upper(extension);
809 
810  if (extension == ".PLY")
811  {
812  reader = vtkPLYReader::New();
813  }
814  else if (extension == ".OBJ")
815  {
816  reader = vtkSmartPointer<vtkOBJReader>::New();
817  }
818  else if (extension == ".STL")
819  {
820  reader = vtkSmartPointer<vtkSTLReader>::New();
821  }
822 
823  reader->SetFileName(fileName_c_str);
824 
825  // set wait cursor
826  QApplication::setOverrideCursor(Qt::WaitCursor);
827  disableMenus(true);
828  qApp->processEvents(); // optional
829 
830 
831 
832  QProgressDialog readerProgressDialog(
833  QString("Loading mesh. Please wait ..."),
834  QString("Hide"), 0, 100, this);
835 
836  readerProgressDialog.setWindowModality(Qt::WindowModal);
837  readerProgressDialog.setRange(0,0);
838  readerProgressDialog.setMinimumDuration(0);
839  readerProgressDialog.raise();
840  readerProgressDialog.show();
841 
842 
843  VTKAlgorithmWaitThread readerThread(reader);
844  readerThread.start();
845 
846  while(!readerThread.isDone())
847  {
848  QCoreApplication::processEvents();
849  }
850 
851  readerProgressDialog.close();
852 
853  // gather some information and show them on the statusbar
854  qlonglong numberOfFaces = static_cast<qlonglong>(reader->GetOutput()->GetNumberOfPolys());
855  qlonglong numberOfVertices = static_cast<qlonglong>(reader->GetOutput()->GetNumberOfPoints());
856 
857  QFileInfo fileInfo(fileName);
858 
859  vtkSmartPointer<vtkPolyData> mesh = reader->GetOutput();
860 
861  double meshBounds[6];
862  mesh->GetBounds(meshBounds);
863 
864 
865  //////////////////////////////////////////////////////////////////////////////////////////////////
866  /// create sceneobject, transform polydata
867  //////////////////////////////////////////////////////////////////////////////////////////////////
868 
869  // first clean polydata (remove merge duplicate points, and/or remove unused points
870  // and/or remove degenerate cells)
871  vtkNew<vtkCleanPolyData> cleanMeshFilter;
872  cleanMeshFilter->SetInputData(mesh);
873  cleanMeshFilter->Update();
874 
875  vtkNew<vtkTriangleFilter> triangleFilter;
876  triangleFilter->SetInputConnection(cleanMeshFilter->GetOutputPort());
877  triangleFilter->Update();
878 
879 
880  // generate point normals
881  vtkNew<vtkPolyDataNormals> normalGenerator;
882  normalGenerator->SetInputConnection(triangleFilter->GetOutputPort());
883  normalGenerator->ComputePointNormalsOn();
884  normalGenerator->ComputeCellNormalsOn();
885  normalGenerator->ConsistencyOn();
886  normalGenerator->SplittingOn();
887  normalGenerator->AutoOrientNormalsOn();
888  normalGenerator->NonManifoldTraversalOn();
889  normalGenerator->Update();
890 
891 
892  // process mesh data
893 
894  // rotate mesh for 3D printing coordinate system to screen coordinate system
895  vtkNew<vtkTransform> transformRotate;
896  transformRotate->RotateX(-90.0);
897 
898  vtkNew<vtkTransformPolyDataFilter> transformFilter1;
899  transformFilter1->SetInputConnection(normalGenerator->GetOutputPort());
900  transformFilter1->SetTransform(transformRotate);
901  transformFilter1->Update();
902 
903  double bounds[6];
904  transformFilter1->GetOutput()->GetBounds(bounds); // get the bounds as (Xmin, Xmax, Ymin, Ymax, Zmin, Zmax).
905 
906  double length[3] = { fabs(bounds[Constants::VTK_X_MAX] - bounds[Constants::VTK_X_MIN]),
907  fabs(bounds[Constants::VTK_Y_MAX] - bounds[Constants::VTK_Y_MIN]),
908  fabs(bounds[Constants::VTK_Z_MAX] - bounds[Constants::VTK_Z_MIN]) };
909 
910  double maxLen = std::max(length[0], std::max(length[1], length[2]));
911 
912 
913  if(scaleMesh)
914  {
915  // resize to unit cube
916  const double factor = this->scene->getMeshScalesize() / maxLen;
917 
918  vtkNew<vtkTransform> transformScale;
919  transformScale->Scale(factor, factor, factor);
920 
921  vtkNew<vtkTransformPolyDataFilter> transformFilter2;
922  transformFilter2->SetInputConnection(transformFilter1->GetOutputPort());
923  transformFilter2->SetTransform(transformScale);
924  transformFilter2->Update();
925 
926 
927  transformFilter2->GetOutput()->GetBounds(bounds); // get the bounds as (Xmin, Xmax, Ymin, Ymax, Zmin, Zmax).
928 
929 
930  // center
931  vtkNew<vtkTransform> transformTranslate;
932 
933  double offset[3] = { -bounds[Constants::VTK_X_MIN],
934  -bounds[Constants::VTK_Y_MIN],
935  -bounds[Constants::VTK_Z_MIN] };
936  transformTranslate->Translate(offset);
937 
938 
939  vtkNew<vtkTransformPolyDataFilter> transformFilter3;
940  transformFilter3->SetInputConnection(transformFilter2->GetOutputPort());
941  transformFilter3->SetTransform(transformTranslate);
942  transformFilter3->Update();
943 
944  this->scene->setMeshPolyData(transformFilter3->GetOutput());
945  }
946  else
947  {
948  // center
949  vtkNew<vtkTransform> transformTranslate;
950 
951  double offset[3] = { -bounds[Constants::VTK_X_MIN],
952  -bounds[Constants::VTK_Y_MIN],
953  -bounds[Constants::VTK_Z_MIN] };
954  transformTranslate->Translate(offset);
955 
956 
957  vtkNew<vtkTransformPolyDataFilter> transformFilter2;
958  transformFilter2->SetInputConnection(transformFilter1->GetOutputPort());
959  transformFilter2->SetTransform(transformTranslate);
960  transformFilter2->Update();
961 
962  this->scene->setMeshScalesize(maxLen);
963  this->scene->setMeshPolyData(transformFilter2->GetOutput());
964  }
965 
966  vtkNew<vtkPolyDataMapper> meshMapper;
967  meshMapper->SetInputData(this->scene->getMeshPolyData());
968  meshMapper->SetResolveCoincidentTopologyToPolygonOffset();
969  meshMapper->SetResolveCoincidentTopologyLineOffsetParameters(2.0, 0.4);
970  meshMapper->SetResolveCoincidentTopology(1);
971  meshMapper->SetResolveCoincidentTopologyZShift(0.5);
972  meshMapper->SetResolveCoincidentTopologyToShiftZBuffer();
973  meshMapper->Update();
974 
975  // visualize
976  this->scene->getMeshActor()->SetMapper(meshMapper);
977  this->scene->getMeshActor()->GetProperty()->SetRepresentationToWireframe();
978 
979  this->scene->getMeshPolyData()->GetBounds(bounds); // get the bounds as (Xmin, Xmax, Ymin, Ymax, Zmin, Zmax).
980 
981 
982  length[0] = fabs(bounds[Constants::VTK_X_MAX] - bounds[Constants::VTK_X_MIN]);
983  length[2] = fabs(bounds[Constants::VTK_Z_MAX] - bounds[Constants::VTK_Z_MIN]);
984 
985  const double shiftX = (appSettings.buildPlate_x_width / 2.0) - (appSettings.buildPlate_x_width - length[0]) / 2.0; //100.0 - (200.0 - length[0]) / 2.0;
986  const double shiftZ = (appSettings.buildPlate_y_depth / 2.0) - (appSettings.buildPlate_y_depth - length[2]) / 2.0; //100.0 - (200.0 - length[2]) / 2.0;
987 
988 
989 
990  vtkNew<vtkTransform> transformActor;
991  transformActor->Translate(shiftX,
992  0.0,
993  shiftZ);
994  this->scene->getWorkPlaneActor()->SetUserTransform(transformActor);
995  this->scene->setLookAt(vtkVector3d(this->scene->getWorkPlaneActor()->GetPosition()));
996 
997 
998  //////////////////////////////////////////////////////////////////////////////////////////////////
999  /// create voxeldata and cubrille actor
1000  //////////////////////////////////////////////////////////////////////////////////////////////////
1001  this->scene->getMeshPolyData()->GetBounds(bounds);
1002 
1003  this->scene->setVoxelData(vtkSmartPointer<vtkImageData>::New());
1004 
1005  QProgressDialog voxelizerProgressDialog(
1006  QString("voxelizing mesh, please wait..."),
1007  QString("Hide"), 0, 0, this);
1008 
1009  voxelizerProgressDialog.setRange(0,0);
1010  voxelizerProgressDialog.setMinimumDuration(0);
1011  voxelizerProgressDialog.show();
1012 
1013 
1014  int resolution = this->getVoxelizerResolution();
1015  this->scene->setVoxelizeResolution(resolution);
1016 
1017 
1018  std::shared_ptr<Voxelizer> voxelizer = this->getVoxelizer();
1019 
1020  voxelizer->setSampleDimensions(resolution, resolution, resolution);
1021  voxelizer->setInputData(this->scene->getMeshPolyData());
1022  this->scene->setVoxelizer(voxelizer);
1023 
1024 
1025  std::chrono::steady_clock::time_point startTime = std::chrono::steady_clock::now();
1026 
1027  VoxelizerThread voxelizerThread(voxelizer.get());
1028 
1029  voxelizerThread.start();
1030 
1031 
1032  while(!voxelizerThread.isDone())
1033  {
1034  QCoreApplication::processEvents();
1035  }
1036 
1037  this->scene->setVoxelData(voxelizer->getOutput());
1038 
1039 
1040  std::chrono::steady_clock::time_point endTime = std::chrono::steady_clock::now();
1041  std::chrono::duration<double, std::milli> elapsedTime_ms = endTime - startTime;
1042 
1043  //double elapsedTimeInSeconds = timer.elapsed() / 1000.0;
1044  double elapsedTimeInSeconds = elapsedTime_ms.count() / 1000.0;
1045 
1046  ui->statusBar->showMessage(
1047  QStringLiteral("%1 vertices: %2"
1048  " faces: %3 voxelization time: %4 s").arg(fileInfo.fileName())
1049  .arg(numberOfVertices)
1050  .arg(numberOfFaces)
1051  .arg(elapsedTimeInSeconds) );
1052 
1053 
1054  VoxelImageVisualizer visualizer;
1055  visualizer.setOutputSize(this->scene->getMeshScalesize());
1056  visualizer.setInputData(this->scene->getVoxelData());
1057  this->scene->setCuberilleActor(visualizer.getColorOutput());
1058 
1059 
1060  this->scene->setInteriorVoxelsActor(visualizer.getInteriorOutput());
1061  this->scene->getInteriorVoxelsActor()->SetVisibility(0);
1062 
1063  this->scene->setInnerAndOuterVoxelsActor(visualizer.getShellAndInsideOutput());
1064  this->scene->getInnerAndOuterVoxelsActor()->SetVisibility(0);
1065 
1066  this->scene->setOccupationMapActor(visualizer.getOccupationOutput());
1067  this->scene->getOccupationMapActor()->SetVisibility(0);
1068 
1069 
1070  this->scene->getCuberilleActor()->GetBounds(bounds);
1071 
1072  this->scene->getCuberilleActor()->GetProperty()->EdgeVisibilityOn();
1073 
1074  this->scene->getInteriorVoxelsActor()->GetProperty()->SetOpacity(0.5);
1075  this->scene->getInteriorVoxelsActor()->GetProperty()->EdgeVisibilityOff();
1076 
1077  this->scene->getInnerAndOuterVoxelsActor()->GetProperty()->SetOpacity(0.5);
1078  this->scene->getInnerAndOuterVoxelsActor()->GetProperty()->EdgeVisibilityOff();
1079 
1080  this->scene->getOccupationMapActor()->GetProperty()->SetOpacity(0.5);
1081  this->scene->getOccupationMapActor()->GetProperty()->EdgeVisibilityOff();
1082 
1083  this->scene->getRenderer()->AddActor(this->scene->getMeshActor());
1084  this->scene->getRenderer()->AddActor(this->scene->getCuberilleActor());
1085  this->scene->getRenderer()->AddActor(this->scene->getInteriorVoxelsActor());
1086  this->scene->getRenderer()->AddActor(this->scene->getInnerAndOuterVoxelsActor());
1087  this->scene->getRenderer()->AddActor(this->scene->getOccupationMapActor());
1088 
1089  QApplication::restoreOverrideCursor();
1090  disableMenus(false);
1091 
1092  double shiftValue = this->scene->getMeshScalesize() / 2.0;
1093  this->scene->setLookAt(vtkVector3d(shiftValue, 0.0, shiftValue));
1094
