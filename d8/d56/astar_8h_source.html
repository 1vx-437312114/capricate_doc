105 /// 3D-octile distance heuristic functor
106 struct DiagonalDistanceHeuristic
107 {
108  inline int operator() (const vtkVector3i& a, const vtkVector3i& b)
109  {
110  int x1 = a.GetX();
111  int y1 = a.GetY();
112  int z1 = a.GetZ();
113 
114  int x2 = b.GetX();
115  int y2 = b.GetY();
116  int z2 = b.GetZ();
117 
118  int dx = abs(x1 - x2);
119  int dy = abs(y1 - y2);
120  int dz = abs(z1 - z2);
121 
122  int delta[3] = { dx, dy, dz };
123 
124  // sort array in increasing order
125  if (delta[0] > delta[1])
126  std::swap(delta[0], delta[1]);
127 
128  if (delta[0] > delta[2])
129  std::swap(delta[0], delta[2]);
130 
131  if (delta[1] > delta[2])
132  std::swap(delta[1], delta[2]);
133 
134  // h = ∆x + ∆y + ∆z − (3 − √3) ∗ delta[0] − (2 − √2) ∗ delta[1]
135  return static_cast<int>(dx + dy + dz - (3.0 - sqrt(3.0) * delta[0]) - (2.0 - sqrt(2.0) * delta[1]));
136  }
137 };

320 /**
321  * @class AStar
322  * @brief Implementation of the A* search algorithm.
323  *
324  * A* is an informed search algorithm. It starts from a specific starting node of a graph and
325  * aims to find a path to the given goal node having the smallest cost (least distance travelled)
326  * It does this by maintaining a tree of paths originating at the start node and extending those
327  * paths one edge at a time until its termination criterion is fullfilled.
328  *
329  * At each iteration of its main loop it needs to determine which of its paths to extend.
330  * It does this based on the cost of the path and an estimate of the cost using a heuristic required to extend
331  * the path all the way to the goal. Specifically, A* selects the path that minimizes
332  * f(n) = g(n) + h(n)
333  * where n is the next node on the path, g(n) is the cost of the path from the start node to n
334  * and h(n) is a heuristic function that estimates the cost of the cheapest path from n to the goal.
335  *
336  * A* terminates when the path it chooses to extend is a path from start to goal or
337  * if there are no paths eligible to be extended.
338  *
339  * The heuristic function is problem-specific. If the heuristic function is admissible, meaning that
340  * it never overestimates the actual cost to get to the goal, A* is guaranteed to return a least-cost
341  * path from start to goal.
342  *
343  * @see https://en.wikipedia.org/wiki/A*_search_algorithm
344  */
