43  loopSplit(Edge *_e, int md)
44  {
45  e = _e;
46  if (e->t1 == NULL || e->t2 == NULL || md) p = ((*e->v1)+(*e->v2))/2.0;
47  else
48  {
49  Vertex *ov1 = e->t1->oppositeVertex(e);

56 double subsurfbeta_loop(int k)
57 {
58  double beta = (cos((2.0*M_PI)/((double)k))/4.0)+(3.0/8.0);
59  return ((5.0/8.0)-(beta*beta))/((double)k);

63 void loopRelaxOriginal(Vertex *v)
64 {
65  Node *n;
66  Edge *e;
67  List *ve;
68  Point np;
69  int k;
70  double beta;
71 
72  if (v->isOnBoundary())
73  {
74  ve = v->VE();
75  np = (*(((Edge *)ve->head()->data)->oppositeVertex(v)));
76  np += (*(((Edge *)ve->tail()->data)->oppositeVertex(v)));
77  np = (((*v)*6.0)+np)/8.0;
78  delete(ve);
79  }
80  else
81  {
82  ve = v->VE();
83  k = ve->numels();
84  beta = subsurfbeta_loop(k);
85  FOREACHVEEDGE(ve, e, n) np += (*(e->oppositeVertex(v)));
86  np = ((*v)*(1.0-k*beta))+(np*beta);
87  delete(ve);
88  }
89 
90  v->info = new Point(&np);
91 }
