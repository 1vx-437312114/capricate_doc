262 /// @brief plane box overlap based on the separating axis theorem
263 /// Adapted from
264 ///
265 /// @cite "Fast 3D triangle-box overlap testing" by
266 /// Tomas Akenine-Möller
267 /// Publication:SIGGRAPH 05: ACM SIGGRAPH 2005 Courses
268 /// July 2005 Pages 8–es
269 /// https://doi.org/10.1145/1198555.1198747
270 ///
271 inline int VoxelOctree::planeBoxOverlap(double normal[3], double vert[3], const double maxbox[3])
272 {
273  double vmin[3], vmax[3];
274 
275  for( int i = 0; i < 3 ; i++ )
276  {
277  double v = vert[i];
278 
279  if(normal[i] > 0.0 )
280  {
281  vmin[i] = -maxbox[i] - v;
282  vmax[i] = maxbox[i] - v;
283  }
284  else
285  {
286  vmin[i] = maxbox[i] - v;
287  vmax[i] = -maxbox[i] - v;
288  }
289  }
290 
291 
292  double dot1 = normal[0] * vmin[0] + normal[1] * vmin[1] + normal[2] * vmin[2];
293  if (dot1 > 0.0)
294  return 0;
295 
296 
297  double dot2 = normal[0] * vmax[0] + normal[1] * vmax[1] + normal[2] * vmax[2];
298  if (dot2 >= 0.0)
299  return 1;
300 
301  return 0;
302 }

313 inline int VoxelOctree::triBoxOverlap(const vtkVector3d& boxcenter, const vtkVector3d& boxHalfsize,
314  const vtkVector3d& vertex0, const vtkVector3d& vertex1, const vtkVector3d& vertex2)
315 {
316  double boxhalfsize[3];
317  boxhalfsize[0] = boxHalfsize[0];
318  boxhalfsize[1] = boxHalfsize[1];
319  boxhalfsize[2] = boxHalfsize[2];
320 
321 
322  double v[3][3];
323  v[0][0] = vertex0[0] - boxcenter[0];
324  v[0][1] = vertex0[1] - boxcenter[1];
325  v[0][2] = vertex0[2] - boxcenter[2];
326 
327 
328  v[1][0] = vertex1[0] - boxcenter[0];
329  v[1][1] = vertex1[1] - boxcenter[1];
330  v[1][2] = vertex1[2] - boxcenter[2];
331 
332 
333  v[2][0] = vertex2[0] - boxcenter[0];
334  v[2][1] = vertex2[1] - boxcenter[1];
335  v[2][2] = vertex2[2] - boxcenter[2];
336 
337 
338 
339  double e[3][3];
340  e[0][0] = v[1][0] - v[0][0];
341  e[0][1] = v[1][1] - v[0][1];
342  e[0][2] = v[1][2] - v[0][2];
343 
344 
345  e[1][0] = v[2][0] - v[1][0];
346  e[1][1] = v[2][1] - v[1][1];
347  e[1][2] = v[2][2] - v[1][2];
348 
349  e[2][0] = v[0][0] - v[2][0];
350  e[2][1] = v[0][1] - v[2][1];
351  e[2][2] = v[0][2] - v[2][2];
352 
353 
354  double fex = fabs(e[0][0]);
355  double fey = fabs(e[0][1]);
356  double fez = fabs(e[0][2]);
357 
358  double p0 = e[0][2] * v[0][1] - e[0][1] * v[0][2];
359  double p2 = e[0][2] * v[2][1] - e[0][1] * v[2][2];
360 
361  double min = std::min(p0, p2);
362  double max = std::max(p0, p2);
363 
364  double rad = fez * boxhalfsize[1] + fey * boxhalfsize[2];
365  if ( min > rad || max < -rad ) return 0;
366 
367  p0 = -e[0][2] * v[0][0] + e[0][0] * v[0][2];
368  p2 = -e[0][2] * v[2][0] + e[0][0] * v[2][2];
369  min = std::min(p0, p2);
370  max = std::max(p0, p2);
371  rad = fez * boxhalfsize[0] + fex * boxhalfsize[2];
372  if ( min > rad || max < -rad ) return 0;
373 
374  double p1 = e[0][1] * v[1][0] - e[0][0] * v[1][1];
375  p2 = e[0][1] * v[2][0] - e[0][0] * v[2][1];
376  min = std::min(p1, p2);
377  max = std::max(p1, p2);
378  rad = fey * boxhalfsize[0] + fex * boxhalfsize[1];
379  if ( min > rad || max < -rad ) return 0;
380 
381  fex = fabs(e[1][0]);
382  fey = fabs(e[1][1]);
383  fez = fabs(e[1][2]);
384 
385  p0 = e[1][2] * v[0][1] - e[1][1] * v[0][2];
386  p2 = e[1][2] * v[2][1] - e[1][1] * v[2][2];
387  min = std::min(p0, p2);
388  max = std::max(p0, p2);
389  rad = fez * boxhalfsize[1] + fey * boxhalfsize[2];
390  if ( min > rad || max < -rad ) return 0;
391 
392  p0 = -e[1][2] * v[0][0] + e[1][0] * v[0][2];
393  p2 = -e[1][2] * v[2][0] + e[1][0] * v[2][2];
394  min = std::min(p0, p2);
395  max = std::max(p0, p2);
396  rad = fez * boxhalfsize[0] + fex * boxhalfsize[2];
397  if ( min > rad || max < -rad ) return 0;
398 
399  p0 = e[1][1] * v[0][0] - e[1][0] * v[0][1];
400  p1 = e[1][1] * v[1][0] - e[1][0] * v[1][1];
401  min = std::min(p0, p1);
402  max = std::max(p0, p1);
403  rad = fey * boxhalfsize[0] + fex * boxhalfsize[1];
404  if ( min > rad || max < -rad ) return 0;
405 
406  fex = fabs(e[2][0]);
407  fey = fabs(e[2][1]);
408  fez = fabs(e[2][2]);
409 
410  p0 = e[2][2] * v[0][1] - e[2][1] * v[0][2];
411  p1 = e[2][2] * v[1][1] - e[2][1] * v[1][2];
412  min = std::min(p0, p1);
413  max = std::max(p0, p1);
414  rad = fez * boxhalfsize[1] + fey * boxhalfsize[2];
415  if ( min > rad || max < -rad ) return 0;
416 
417  p0 = -e[2][2] * v[0][0] + e[2][0] * v[0][2];
418  p1 = -e[2][2] * v[1][0] + e[2][0] * v[1][2];
419  min = std::min(p0, p1);
420  max = std::max(p0, p1);
421  rad = fez * boxhalfsize[0] + fex * boxhalfsize[2];
422  if ( min > rad || max < -rad ) return 0;
423 
424  p1 = e[2][1]*v[1][0] - e[2][0] * v[1][1];
425  p2 = e[2][1]*v[2][0] - e[2][0] * v[2][1];
426  min = std::min(p1, p2);
427  max = std::max(p1, p2);
428  rad = fey * boxhalfsize[0] + fex * boxhalfsize[1];
429  if ( min > rad || max < -rad ) return 0;
430 
431 
432  for ( int i = 0; i < 3; i++ )
433  {
434  min = std::min(std::min(v[0][i], v[1][i]), v[2][i]);
435  max = std::max(std::max(v[0][i], v[1][i]), v[2][i]);
436 
437  if ( min > boxhalfsize[i] || max < -boxhalfsize[i] )
438  return 0;
439  }
440 
441  double normal[3];
442  normal[0] = e[0][1] * e[1][2] - e[0][2] * e[1][1];
443  normal[1] = e[0][2] * e[1][0] - e[0][0] * e[1][2];
444  normal[2] = e[0][0] * e[1][1] - e[0][1] * e[1][0];
445 
446  if ( !planeBoxOverlap(normal, v[0], boxhalfsize) )
447  return 0;
448 
449  return 1;
450 }

474 /// Computes the closest point from a position to a triangle
475 inline vtkVector3d VoxelOctree::closestPointOnTriangle(vtkVector3d triangleVertices[3], const vtkVector3d& pos)
476 {
477  vtkVector3d edge0 = triangleVertices[1] - triangleVertices[0];
478  vtkVector3d edge1 = triangleVertices[2] - triangleVertices[0];
479  vtkVector3d v0 = triangleVertices[0] - pos;
480 
481 
482  double a = vtkMath::Dot(edge0.GetData(), edge0.GetData());
483  double b = vtkMath::Dot(edge0.GetData(), edge1.GetData());
484  double c = vtkMath::Dot(edge1.GetData(), edge1.GetData());
485  double d = vtkMath::Dot(edge0.GetData(), v0.GetData());
486  double e = vtkMath::Dot(edge1.GetData(), v0.GetData());
487 
488  double det = a * c - b * b;
489  double s = b * e - c * d;
490  double t = b * d - a * e;
491 
492  if (s + t < det)
493  {
494  if (s < 0.0)
495  {
496  if (t < 0.0)
497  {
498  if (d < 0.0)
499  {
500  s = saturate(-d / a);
501  t = 0.0;
502  }
503  else
504  {
505  s = 0.0;
506  t = saturate(-e / c);
507  }
508  }
509  else
510  {
511  s = 0.0;
512  t = saturate(-e / c);
513  }
514  }
515  else if (t < 0.0)
516  {
517  s = saturate(-d / a);
518  t = 0.0;
519  }
520  else
521  {
522  double invDet = 1.0 / det;
523  s *= invDet;
524  t *= invDet;
525  }
526  }
527  else
528  {
529  if (s < 0.0)
530  {
531  double tmp0 = b + d;
532  double tmp1 = c + e;
533 
534  if (tmp1 > tmp0)
535  {
536  double numer = tmp1 - tmp0;
537  double denom = a - 2.0 * b + c;
538 
539  s = saturate(numer / denom);
540  t = 1.0 - s;
541  }
542  else
543  {
544  t = saturate(-e / c);
545  s = 0.0;
546  }
547  }
548  else if (t < 0.0)
549  {
550  if (a + d > b + e)
551  {
552  double numer = c + e - b - d;
553  double denom = a - 2.0 * b + c;
554 
555  s = saturate(numer / denom);
556  t = 1.0 - s;
557  }
558  else
559  {
560  s = saturate(-e / c);
561  t = 0.0;
562  }
563  }
564  else
565  {
566  double numer = c + e - b - d;
567  double denom = a - 2.0 * b + c;
568 
569  s = saturate(numer / denom);
570  t = 1.0 - s;
571  }
572  }
573 
574  return triangleVertices[0] + s * edge0 + t * edge1;
575 }

578 /// @brief Plane box overlap based on the separating axis theorem
579 ///
580 /// Adapted from
581 ///
582 /// @cite "An efficient and robust ray-box intersection algorithm" by
583 /// Amy Williams, Steve Barrus, R Keith Morley, Peter Shirley
584 ///
585 /// Publication: SIGGRAPH '05: ACM SIGGRAPH 2005 Courses
586 /// July 2005 Pages 9–es
587 /// https://doi.org/10.1145/1198555.1198748
588 ///
589 /// \param normal Vertex normal
590 /// \param vert Vertex position
591 /// \param maxbox AABB maximal bounds
592 /// \return 0 -> no overlap, 1 -> overlap
593 inline bool VoxelOctree::rayBoxIntersection(const Ray& ray, const AABBd& aabb)
594 {
595  double l1 = (aabb.min.GetX() - ray.origin.GetX()) * ray.invDirection.GetX();
596  double l2 = (aabb.max.GetX() - ray.origin.GetX()) * ray.invDirection.GetX();
597 
598  double tmin = std::min(l1, l2);
599  double tmax = std::max(l1, l2);
600 
601  l1 = (aabb.min.GetY() - ray.origin.GetY()) * ray.invDirection.GetY();
602  l2 = (aabb.max.GetY() - ray.origin.GetY()) * ray.invDirection.GetY();
603 
604  tmin = std::max(std::min(l1, l2), tmin);
605  tmax = std::min(std::max(l1, l2), tmax);
606 
607  l1 = (aabb.min.GetZ() - ray.origin.GetZ()) * ray.invDirection.GetZ();
608  l2 = (aabb.max.GetZ() - ray.origin.GetZ()) * ray.invDirection.GetZ();
609 
610  tmin = std::max(std::min(l1, l2), tmin);
611  tmax = std::min(std::max(l1, l2), tmax);
612  tmin = std::max(0.0, tmin);
613 
614  return (tmax > tmin);
615 }
