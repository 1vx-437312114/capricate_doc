134 /// @brief plane box overlap based on the separating axis theorem
135 /// Adapted from
136 ///
137 /// @cite "Fast 3D triangle-box overlap testing" by
138 /// Tomas Akenine-Möller
139 /// Publication:SIGGRAPH 05: ACM SIGGRAPH 2005 Courses
140 /// July 2005 Pages 8–es
141 /// https://doi.org/10.1145/1198555.1198747
142 ///
143 /// \param normal Vertex normal
144 /// \param vert Vertex position
145 /// \param maxbox AABB maximal bounds
146 /// \return 0 -> no overlap, 1 -> overlap
147 inline int planeBoxOverlap(double normal[3], double vert[3], const double maxbox[3])
148 {
149  double vmin[3], vmax[3];
150 
151  for( int i = 0; i < 3 ; i++ )
152  {
153  double v = vert[i];
154 
155  if(normal[i] > 0.0 )
156  {
157  vmin[i] = -maxbox[i] - v;
158  vmax[i] = maxbox[i] - v;
159  }
160  else
161  {
162  vmin[i] = maxbox[i] - v;
163  vmax[i] = -maxbox[i] - v;
164  }
165  }
166 
167 
168  double dot1 = normal[0] * vmin[0] + normal[1] * vmin[1] + normal[2] * vmin[2];
169  if (dot1 > 0.0)
170  return 0;
171 
172 
173  double dot2 = normal[0] * vmax[0] + normal[1] * vmax[1] + normal[2] * vmax[2];
174  if (dot2 >= 0.0)
175  return 1;
176 
177  return 0;
178 }

181 // based on Moeller Tri-Box Overlap Test
182 inline int triBoxOverlap(const vtkVector3d& boxcenter, const vtkVector3d& boxHalfsize,
183  const vtkVector3d& vertex0, const vtkVector3d& vertex1, const vtkVector3d& vertex2)
184 {
185  double boxhalfsize[3];
186  boxhalfsize[0] = boxHalfsize[0];
187  boxhalfsize[1] = boxHalfsize[1];
188  boxhalfsize[2] = boxHalfsize[2];
189 
190 
191  double v[3][3];
192  v[0][0] = vertex0[0] - boxcenter[0];
193  v[0][1] = vertex0[1] - boxcenter[1];
194  v[0][2] = vertex0[2] - boxcenter[2];
195 
196 
197  v[1][0] = vertex1[0] - boxcenter[0];
198  v[1][1] = vertex1[1] - boxcenter[1];
199  v[1][2] = vertex1[2] - boxcenter[2];
200 
201 
202  v[2][0] = vertex2[0] - boxcenter[0];
203  v[2][1] = vertex2[1] - boxcenter[1];
204  v[2][2] = vertex2[2] - boxcenter[2];
205 
206 
207 
208  double e[3][3];
209  e[0][0] = v[1][0] - v[0][0];
210  e[0][1] = v[1][1] - v[0][1];
211  e[0][2] = v[1][2] - v[0][2];
212 
213 
214  e[1][0] = v[2][0] - v[1][0];
215  e[1][1] = v[2][1] - v[1][1];
216  e[1][2] = v[2][2] - v[1][2];
217 
218  e[2][0] = v[0][0] - v[2][0];
219  e[2][1] = v[0][1] - v[2][1];
220  e[2][2] = v[0][2] - v[2][2];
221 
222 
223  double fex = fabs(e[0][0]);
224  double fey = fabs(e[0][1]);
225  double fez = fabs(e[0][2]);
226 
227  double p0 = e[0][2] * v[0][1] - e[0][1] * v[0][2];
228  double p2 = e[0][2] * v[2][1] - e[0][1] * v[2][2];
229 
230  double min = std::min(p0, p2);
231  double max = std::max(p0, p2);
232 
233  double rad = fez * boxhalfsize[1] + fey * boxhalfsize[2];
234  if ( min > rad || max < -rad ) return 0;
235 
236  p0 = -e[0][2] * v[0][0] + e[0][0] * v[0][2];
237  p2 = -e[0][2] * v[2][0] + e[0][0] * v[2][2];
238  min = std::min(p0, p2);
239  max = std::max(p0, p2);
240  rad = fez * boxhalfsize[0] + fex * boxhalfsize[2];
241  if ( min > rad || max < -rad ) return 0;
242 
243  double p1 = e[0][1] * v[1][0] - e[0][0] * v[1][1];
244  p2 = e[0][1] * v[2][0] - e[0][0] * v[2][1];
245  min = std::min(p1, p2);
246  max = std::max(p1, p2);
247  rad = fey * boxhalfsize[0] + fex * boxhalfsize[1];
248  if ( min > rad || max < -rad ) return 0;
249 
250  fex = fabs(e[1][0]);
251  fey = fabs(e[1][1]);
252  fez = fabs(e[1][2]);
253 
254  p0 = e[1][2] * v[0][1] - e[1][1] * v[0][2];
255  p2 = e[1][2] * v[2][1] - e[1][1] * v[2][2];
256  min = std::min(p0, p2);
257  max = std::max(p0, p2);
258  rad = fez * boxhalfsize[1] + fey * boxhalfsize[2];
259  if ( min > rad || max < -rad ) return 0;
260 
261  p0 = -e[1][2] * v[0][0] + e[1][0] * v[0][2];
262  p2 = -e[1][2] * v[2][0] + e[1][0] * v[2][2];
263  min = std::min(p0, p2);
264  max = std::max(p0, p2);
265  rad = fez * boxhalfsize[0] + fex * boxhalfsize[2];
266  if ( min > rad || max < -rad ) return 0;
267 
268  p0 = e[1][1] * v[0][0] - e[1][0] * v[0][1];
269  p1 = e[1][1] * v[1][0] - e[1][0] * v[1][1];
270  min = std::min(p0, p1);
271  max = std::max(p0, p1);
272  rad = fey * boxhalfsize[0] + fex * boxhalfsize[1];
273  if ( min > rad || max < -rad ) return 0;
274 
275  fex = fabs(e[2][0]);
276  fey = fabs(e[2][1]);
277  fez = fabs(e[2][2]);
278 
279  p0 = e[2][2] * v[0][1] - e[2][1] * v[0][2];
280  p1 = e[2][2] * v[1][1] - e[2][1] * v[1][2];
281  min = std::min(p0, p1);
282  max = std::max(p0, p1);
283  rad = fez * boxhalfsize[1] + fey * boxhalfsize[2];
284  if ( min > rad || max < -rad ) return 0;
285 
286  p0 = -e[2][2] * v[0][0] + e[2][0] * v[0][2];
287  p1 = -e[2][2] * v[1][0] + e[2][0] * v[1][2];
288  min = std::min(p0, p1);
289  max = std::max(p0, p1);
290  rad = fez * boxhalfsize[0] + fex * boxhalfsize[2];
291  if ( min > rad || max < -rad ) return 0;
292 
293  p1 = e[2][1]*v[1][0] - e[2][0] * v[1][1];
294  p2 = e[2][1]*v[2][0] - e[2][0] * v[2][1];
295  min = std::min(p1, p2);
296  max = std::max(p1, p2);
297  rad = fey * boxhalfsize[0] + fex * boxhalfsize[1];
298  if ( min > rad || max < -rad ) return 0;
299 
300 
301  for ( int i = 0; i < 3; i++ )
302  {
303  min = std::min(std::min(v[0][i], v[1][i]), v[2][i]);
304  max = std::max(std::max(v[0][i], v[1][i]), v[2][i]);
305 
306  if ( min > boxhalfsize[i] || max < -boxhalfsize[i] )
307  return 0;
308  }
309 
310  double normal[3];
311  normal[0] = e[0][1] * e[1][2] - e[0][2] * e[1][1];
312  normal[1] = e[0][2] * e[1][0] - e[0][0] * e[1][2];
313  normal[2] = e[0][0] * e[1][1] - e[0][1] * e[1][0];
314 
315  if ( !planeBoxOverlap(normal, v[0], boxhalfsize) )
316  return 0;
317 
318  return 1;
319 }
