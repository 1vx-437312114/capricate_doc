73 int vtxEdgeCompare(const void *a, const void *b)
74 {
75  Vertex *va1 = ((Edge *)a)->v1;
76  Vertex *va2 = ((Edge *)a)->v2;
77  Vertex *vb1 = ((Edge *)b)->v1;
78  Vertex *vb2 = ((Edge *)b)->v2;
79  Vertex *tmp;
80 
81  if (va2<va1) { tmp = va1; va1 = va2; va2 = tmp; }
82  if (vb2<vb1) { tmp = vb1; vb1 = vb2; vb2 = tmp; }
83  if (va1<vb1) return -1;
84  if (va1>vb1) return 1;
85  if (va2<vb2) return -1;
86  if (va2>vb2) return 1;
87  return 0;

183  Triangle *ta2 = (e2 != NULL)?(e2->oppositeTriangle(t1)):(NULL);
184  Triangle *ta3 = (e3 != NULL)?(e3->oppositeTriangle(t2)):(NULL);
185  Triangle *ta4 = (e4 != NULL)?(e4->oppositeTriangle(t2)):(NULL);
186 
187  if (v1->isOnBoundary() && v2->isOnBoundary())
188  if (!(((ta1 || ta2) && !ta3 && !ta4) || ((ta3 || ta4) && !ta1 && !ta2))) return NULL;
189 
190  if (ta1 != NULL && ta2 != NULL && ta1->oppositeVertex(e1) == ta2->oppositeVertex(e2))
191  return NULL;
192  if (ta3 != NULL && ta4 != NULL && ta3->oppositeVertex(e3) == ta4->oppositeVertex(e4))
193  return NULL;
194 
195  if (ta1 == NULL && ta2 == NULL) v1->e0 = e3;
196  else v1->e0 = e2;
197 
198  if (v3 != NULL) v3->e0 = e2;
199  if (v4 != NULL) v4->e0 = e3;
200 
201  ve = v2->VE();
202  FOREACHVEEDGE(ve, e, n)
203  {
204  tv = e->oppositeVertex(v2);
205  if (tv != v3 && tv != v4 && tv->getEdge(v1) != NULL) {delete(ve); return NULL;}
206  }
207  FOREACHVEEDGE(ve, e, n) if (e != this) e->replaceVertex(v2, v1);
208  delete(ve);
209 
210  if (e2 != NULL) e2->replaceTriangle(t1, ta1);
211  if (e3 != NULL) e3->replaceTriangle(t2, ta4);
212 
213  if (ta1 != NULL) ta1->replaceEdge(e1, e2);
214  if (ta4 != NULL) ta4->replaceEdge(e4, e3);
215 
216  v2->e0 = NULL; // v2 must be removed
217  if (e4 != NULL) e4->v1 = e4->v2 = NULL; // e4 must be removed
218  if (e1 != NULL) e1->v1 = e1->v2 = NULL; // e1 must be removed
219  if (t1 != NULL) t1->e1 = t1->e2 = t1->e3 = NULL; // t1 must be removed
220  if (t2 != NULL) t2->e1 = t2->e2 = t2->e3 = NULL; // t2 must be removed
221 
222  if (e2 != NULL && e2->t1 == NULL && e2->t2 == NULL)
223  {
224  v3->e0 = NULL;
225  e2->v1 = e2->v2 = NULL;
226  }
227  if (e3 != NULL && e3->t1 == NULL && e3->t2 == NULL)
228  {
229  v4->e0 = NULL;
230  e3->v1 = e3->v2 = NULL;
231  }
232

234

236

344 double Edge::delaunayMinAngle() const
345 {
346  if (t1==NULL || t2==NULL) return 2*M_PI;
347  if (squaredLength()==0) return 0;
348  if (t1->nextEdge(this)->squaredLength() == 0) return 0;
349  if (t1->prevEdge(this)->squaredLength() == 0) return 0;
350  double a1 = t1->getAngle(v1);
351  double a2 = t1->getAngle(v2);
352  double a3 = t1->getAngle(t1->oppositeVertex(this));
353  if (t2->nextEdge(this)->length()==0) return 0;
354  if (t2->prevEdge(this)->length()==0) return 0;
355  double a4 = t2->getAngle(v1);
356  double a5 = t2->getAngle(v2);
357  double a6 = t2->getAngle(t2->oppositeVertex(this));
358 
359  if (a1+a4 >= M_PI || a2+a5 >= M_PI) return 3*M_PI;
360  return MIN(a1,(MIN(a2,(MIN(a3,(MIN(a4,(MIN(a5,a6)))))))));
361 }
